1 /** 
  2  * オブジェクトの値を表す文字列を返します。
  3  * @return {String} オブジェクトの値を表す文字列
  4  */
  5 Object.prototype.toString = function() {
  6   var result = "{";
  7   for(e in this) {
  8     if(this.hasOwnProperty(e)) {
  9       if(result !== "{") {
 10         result += ", ";
 11       }
 12       result += '"' + e + '": ' + this[e].toString();
 13     }
 14   }
 15   result += "}";
 16   return result;
 17 };

 19 /**
 20  * pをプロトタイプに持つオブジェクトを生成します。
 21  * @param {Object} p プロトタイプ
 22  * @return {Object} pをプロトタイプに持つオブジェクト
 23  */
 24 Object.prototype.create = function(p) {
 25   function f() {};
 26   f.prototype = p;
 27   return new f();
 28 };
