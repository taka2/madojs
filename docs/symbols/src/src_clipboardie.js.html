1 /** 
  2  * 新しいクリップボードを作成する
  3  * @class InternetExplorerを利用した、クリップボードへの格納、および、取得を行うクラス。
  4  */
  5 var ClipboardIE = function() {
  6   this._ie = new ActiveXObject('InternetExplorer.Application');
  7   this._ie.navigate("about:blank");
  8   while(this._ie.Busy) {
  9     sleep(10);
 10   }
 11   this._ie.Visible = false;
 12   this._textarea = this._ie.document.createElement("textarea");
 13   this._ie.document.body.appendChild(this._textarea);
 14   this._textarea.focus();
 15   this._closed = false;
 16 };
 17 
 18 // Prototypes of ClipboardIE
 19 ClipboardIE.prototype = {
 20   /**
 21    * クリップボードへ文字列を格納します。
 22    * @param {String} text 格納する文字列
 23    */
 24   set: function(text) {
 25     if (this._closed) {
 26       return;
 27     }
 28     this._textarea.innerText = text;
 29     this._ie.execWB(17 /* select all */, 0);
 30     this._ie.execWB(12 /* copy */, 0);
 31   },
 32   /**
 33    * クリップボードの文字列を取得します。
 34    * @return {String} 取得した文字列
 35    */
 36   get: function() {
 37     if (this._closed) {
 38       return;
 39     }
 40     this._textarea.innerText = "";
 41     this._ie.execWB(13 /* paste */, 0);
 42     return this._textarea.innerText;
 43   },
 44   /**
 45    * クリップボードをクローズします。
 46    */
 47   close: function() {
 48     this._ie.Quit();
 49     this._closed = true;
 50   }
 51 };
