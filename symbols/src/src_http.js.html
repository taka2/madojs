1 /** 
  2  * hostとportを指定してHTTPオブジェクトを作成します。
  3  * @class HTTPの操作を行うためのクラスです。
  4 <pre class = "code">
  5 使用例：
  6 HTTP.start("www.yahoo.co.jp", 80, function(http) {
  7   print(http.get("/index.html"));
  8 });
  9 </pre>
 10  * @param {String} host HTTP操作の対象ホスト
 11  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 12  */
 13 var HTTP = function(host, port) {
 14   this.host = host;
 15   this.port = port || DEFAULT_PORT_NUMBER;
 16 };
 17 
 18 /**
 19  * 定数：デフォルトのポート番号(80)
 20  */
 21 HTTP.DEFAULT_PORT_NUMBER = 80;
 22 
 23 /** 
 24  * hostとportを指定してHTTPオブジェクトを作成し、ブロックを実行します。
 25  * ブロックが指定されていない場合は、作成したHTTPオブジェクトを返します。
 26  * @param {String} host HTTP操作の対象ホスト
 27  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 28  * @param {Function} block (オプション)ブロック
 29  * @return {Object} ブロックが指定されていない場合は、作成したHTTPオブジェクト
 30  */
 31 HTTP.start = function(host, port, block) {
 32   var http = new HTTP(host, port);
 33   if(block) {
 34     block(http);
 35   } else {
 36     return http;
 37   }
 38 };
 39 
 40 /** 
 41  * 指定されたaddress、path、portに対して、GETリクエストを行います。
 42  * @param {String} address HTTP操作の対象ホスト
 43  * @param {String} path HTTP操作の対象パス
 44  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 45  * @return {String} レスポンステキスト
 46  * @example 使用例：
 47 print(HTTP.get("www.yahoo.co.jp", "/index.html", 80));
 48  */
 49 HTTP.get = function(address, path, port) {
 50   var myPort = port || DEFAULT_PORT_NUMBER;
 51   var http = new HTTP(address, port);
 52   return http.get(path);
 53 };
 54 
 55 /** 
 56  * 指定されたaddress、path、portに対して、GETリクエストを行い、printします。
 57  * @param {String} address HTTP操作の対象ホスト
 58  * @param {String} path HTTP操作の対象パス
 59  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 60  * @example 使用例：
 61 HTTP.get_print("www.yahoo.co.jp", "/index.html", 80);
 62  */
 63 HTTP.get_print = function(address, path, port) {
 64   print(HTTP.get(address, path, port));
 65 };
 66 
 67 // Prototypes of HTTP
 68 HTTP.prototype = {
 69   /** 
 70    * 指定されたpath、headerでGETリクエストを行い、ブロックを実行します。
 71    * ブロックが指定されていない場合は、レスポンステキストを返します。
 72    * @param {String} path HTTP操作の対象パス
 73    * @param {Object} header (オプション)HTTPヘッダ
 74    * @param {Function} block (オプション)ブロック
 75    * @return {String} ブロックが指定されていない場合は、レスポンステキスト
 76    * @throws 200OK以外のレスポンスが返された場合にスローされます。
 77    */
 78   get: function(path, header, block) {
 79     return this.request("GET", path, header, "", block);
 80   },
 81   /** 
 82    * 指定されたpath、headerでPOSTリクエストを行い、ブロックを実行します。
 83    * ブロックが指定されていない場合は、レスポンステキストを返します。
 84    * @param {String} path HTTP操作の対象パス
 85    * @param {Object} header (オプション)HTTPヘッダ(初期値 [])
 86    * @param {Object} body (オプション)HTTPボディ(初期値 undefined)
 87    * @param {Function} block (オプション)ブロック
 88    * @return {String} ブロックが指定されていない場合は、レスポンステキスト
 89    * @throws 200OK以外のレスポンスが返された場合にスローされます。
 90    */
 91   post: function(path, header, body, block) {
 92     return this.request("POST", path, header, body, block);
 93   },
 94   // private
 95   request: function(method, path, header, body, block) {
 96     // パラメータの調整
 97     var myHeader = header || [];
 98 
 99     // リクエストの準備
100     var xhr = new ActiveXObject("Microsoft.XMLHTTP");
101     xhr.open(method, "http://" + this.host + ":" + this.port + path, false);
102 
103     for(var headerName in myHeader) {
104       xhr.setRequestHeader(headerName, myHeader[headerName]);
105     }
106 
107     // リクエストの送信
108     xhr.send(body);
109 
110     // レスポンスの処理
111     if(xhr.readyState === 4) {
112       if(xhr.status === 200) {
113         if(isFunction(block)) {
114           block(xhr.responseText);
115         } else {
116           return xhr.responseText;
117         }
118       } else {
119         throw new Error(xhr.statusText);
120       }
121     }
122   }
123 };
