1 /** 
  2  * Excelオブジェクトを作成します。
  3  * @class Excelファイルの操作と読み書きを行うためのクラスです。<br/>
  4  * Excelがインストールされている必要があります。
  5 <pre class = "code">
  6 使用例：
  7 // abc.xlsを開いて、Sheet1の、1行1列目の値を'hoge'に書き換えつつ、上書き保存する。
  8 Excel.open("abc.xls", function(excel) {
  9   var sheet = excel.getSheetByName("Sheet1");
 10   sheet.setValue(1, 1, 'hoge');
 11   excel.save();
 12 });
 13 </pre>
 14  * @param {String} path Excelファイルのパスを文字列で指定します。
 15  * @throws ファイルが存在しない場合にスローされます。
 16  */
 17 var Excel = function(path) {
 18   this.path = path;
 19 
 20   if(!File.exist(path)) {
 21     throw new Error(-1, "File Not Found: " + path);
 22   }
 23 
 24   this.excelObj = new ActiveXObject("Excel.Application");
 25   this.workbookObj = this.excelObj.Workbooks.Open(path);
 26 };
 27 
 28 /** 
 29  * Excelファイルを開き、ブロックを実行します。
 30  * ブロックが指定されていない場合は、Excelオブジェクトを返します。
 31  * @param {String} path Excelファイルのパスを文字列で指定します。
 32  * @param {Function} block ブロック
 33  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 34  */
 35 Excel.open = function(path, block) {
 36   if(!isFunction(block)) {
 37     return new Excel(path);
 38   }
 39 
 40   try {
 41     var excel = new Excel(path);
 42     block(excel);
 43   } finally {
 44     if (excel != null) {
 45       excel.quit();
 46     }
 47   }
 48 };
 49 
 50 // Prototypes of Excel
 51 Excel.prototype = {
 52   /**
 53    * ワークブックに含まれる、ワークシートを処理します。
 54    * @param {Function} block ブロック
 55    */
 56   each: function(block) {
 57     var enumSheet = new Enumerator(this.workbookObj.Sheets);
 58     for (;!enumSheet.atEnd(); enumSheet.moveNext()) {
 59        block(new ExcelSheet(enumSheet.item()));
 60     }
 61   },
 62   /**
 63    * 指定したsheetNameのワークシートを取得します。
 64    * @param {String} sheetName 取得するワークシート名
 65    * @return {Object} ワークシートオブジェクト
 66    */
 67   getSheetByName: function(sheetName) {
 68     var result;
 69     this.each(function(sheet) {
 70       if(sheet.getName() === sheetName) {
 71         result = sheet;
 72       }
 73     });
 74 
 75     return result;
 76   },
 77   /** 
 78    * Excelファイルを保存します。
 79    */
 80   save: function() {
 81     this.workbookObj.Save();
 82   },
 83   /** 
 84    * Excelファイルをファイル名を指定して保存します。
 85    */
 86   saveAs: function(fileName) {
 87     this.workbookObj.SaveAs(fileName);
 88   },
 89   /** 
 90    * Excelを終了します。
 91    */
 92   quit: function() {
 93     this.workbookObj.Close();
 94     this.excelObj.Quit();
 95   }
 96 };

 98 /** 
 99  * Excelシートオブジェクトを作成します。
100  * @class Excelシートの操作と読み書きを行うためのクラスです。<br/>
101  * Excelがインストールされている必要があります。
102  * @param {String} sheetObj Worksheetオブジェクトを指定します。
103  */
104 var ExcelSheet = function(sheetObj) {
105   this.sheetObj = sheetObj;
106 };
107 
108 // Prototypes of ExcelSheet
109 ExcelSheet.prototype = {
110   /**
111    * ワークシート名を取得します。
112    * @return {String} ワークシート名
113    */
114   getName: function() {
115     return this.sheetObj.Name;
116   },
117   /**
118    * セルに値を設定します。
119    * @param {Number} x x座標
120    * @param {Number} y y座標
121    * @param {Object} value 設定する値
122    */
123   setValue: function(x, y, value) {
124     this.sheetObj.Cells(x, y).Value = value;
125   },
126   /**
127    * セルの値を取得します。
128    * @param {Number} x x座標
129    * @param {Number} y y座標
130    * @return {Object} セルの値
131    */
132   getValue: function(x, y) {
133     return this.sheetObj.Cells(x, y).Value;
134   }
135 };
