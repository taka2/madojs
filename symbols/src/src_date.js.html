1 /** 
  2  * 日付をYYYYMMDD形式で返します。
  3  * @return {String} YYYYMMDD形式の日付
  4  * @example 使用例：
  5 print(new Date().getYYYYMMDD());
  6  */
  7 Date.prototype.getYYYYMMDD = function() {
  8   var yyyy = this.getFullYear();
  9   var mm = this.getMonth() + 1;
 10   var dd = this.getDate();
 11   if (mm < 10) mm = "0" + mm;
 12   if (dd < 10) dd = "0" + dd;
 13   return "" + yyyy + mm + dd;
 14 };
 15 
 16 /** 
 17  * 日付をYYYYMMDDHH24MISS形式で返します。
 18  * @return {String} YYYYMMDDHH24MISS形式の日付
 19  * @example 使用例：
 20 print(new Date().getYYYYMMDDHH24MISS());
 21  */
 22 Date.prototype.getYYYYMMDDHH24MISS = function() {
 23   var hh = this.getHours();
 24   var mi = this.getMinutes();
 25   var ss = this.getSeconds();
 26   if (hh < 10) hh = "0" + hh;
 27   if (mi < 10) mi = "0" + mi;
 28   if (ss < 10) ss = "0" + ss;
 29   return this.getYYYYMMDD() + hh + mi + ss;
 30 };
 31 
 32 /** 
 33  * 日付をYYYY/MM/DD HH:MI:SS形式で返します。
 34  * @return {String} YYYY/MM/DD HH:MM:SS形式の日付
 35  * @example 使用例：
 36 print(new Date().getFormattedDate());
 37  */
 38 Date.prototype.getFormattedDate = function() {
 39   var yyyy = this.getFullYear();
 40   var mm = this.getMonth() + 1;
 41   var dd = this.getDate();
 42   var hh = this.getHours();
 43   var mi = this.getMinutes();
 44   var ss = this.getSeconds();
 45   if (mm < 10) mm = "0" + mm;
 46   if (dd < 10) dd = "0" + dd;
 47   if (hh < 10) hh = "0" + hh;
 48   if (mi < 10) mi = "0" + mi;
 49   if (ss < 10) ss = "0" + ss;
 50   return yyyy + "/" + mm + "/" + dd + " " + hh + ":" + mi + ":" + ss;
 51 };

 53 /**
 54  * 日付に指定ミリ秒足します。
 55  * @param {Number} msec 日付に足すミリ秒
 56  * @return {Date} 指定ミリ秒足した結果の日付
 57  * @example 使用例：
 58 print(new Date().getFormattedDate());
 59  */
 60 Date.prototype.add = function(msec) {
 61   this.setTime(this.getTime() + msec);
 62   return this;
 63 };
