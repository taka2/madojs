1 /**
  2  * インスタンス化しません。
  3  * @class シェル(explorer)へのアクセスを提供するクラス
  4 <pre class = "code">
  5 使用例：
  6 // Windowsフォルダをエクスプローラで開く
  7 Shell.open("C:\\windows");
  8 </pre>
  9  */
 10 var Shell = {};
 11 
 12 /**
 13  * 指定したpathをexplorerで開きます。
 14  * @param {String} path explorerで開くパス
 15  */
 16 Shell.open = function (path) {
 17   Const.SHELLAPP.Open(path);
 18 };
 19 
 20 /**
 21  * 指定したserviceNameのサービスを開始します。
 22  * @param {String} serviceName 開始するサービス名
 23  * @param {Boolean} isPersistent (オプション)スタートアップの種類を自動に変更します(初期値 false)
 24  * @return {Boolean} 成功した場合はtrue、失敗した場合はfalseを返します。
 25  */
 26 Shell.serviceStart = function(serviceName, isPersistent) {
 27   myIsPersistent = isPersistent || false;
 28   return Const.SHELLAPP.ServiceStart(serviceName, myIsPersistent);
 29 };
 30 
 31 /**
 32  * 指定したserviceNameのサービスを停止します。
 33  * @param {String} serviceName 停止するサービス名
 34  * @param {Boolean} isPersistent (オプション)スタートアップの種類を手動に変更します(初期値 false)
 35  * @return {Boolean} 成功した場合はtrue、失敗した場合はfalseを返します。
 36  */
 37 Shell.serviceStop = function(serviceName, isPersistent) {
 38   myIsPersistent = isPersistent || false;
 39   return Const.SHELLAPP.ServiceStop(serviceName, myIsPersistent);
 40 };
