1 /** 
  2  * ファイルオブジェクトを作成します。
  3  * @class ファイルの操作と読み書きを行うためのクラスです。
  4 <pre class = "code">
  5 使用例：
  6 File.open("copy_of_mado.js", "w", function(outfile) {
  7   File.open("mado.js", "r", function(infile) {
  8     infile.each(function(line) {
  9       outfile.puts(line);
 10     });
 11   });
 12 });
 13 </pre>
 14  * @param {String} path ファイルのパスを文字列で指定します。
 15  * @param {String} mode ファイルを読み書きするためのモードを指定します。
 16  * <ul>
 17  *   <li>'r': 読み取り専用</li>
 18  *   <li>'w': 書き込み専用</li>
 19  *   <li>'a': 追加書き込み</li>
 20  * </ul>
 21  * @throws モードが'r'または'a'のときで、ファイルが存在しない場合にスローされます。
 22  */
 23 var File = function(path, mode) {
 24   this.path = path;
 25   this.mode = mode;
 26 
 27   if(this.mode === 'w') {
 28     this.ts = Const.FSO.CreateTextFile(this.path);
 29   } else if(this.mode === 'a') {
 30     if(File.exist(this.path)) {
 31       this.ts = Const.FSO.OpenTextFile(this.path, Const.IOMODE_FORAPPENDING);
 32     } else {
 33       this.ts = Const.FSO.CreateTextFile(this.path);
 34     }
 35   } else {
 36     this.ts = Const.FSO.OpenTextFile(this.path, Const.IOMODE_FORREADING);
 37   }
 38 };

 40 /** 
 41  * 指定したpathのファイルを指定したmodeで開き、ブロックを実行します。
 42  * ブロックが指定されていない場合は、作成したファイルオブジェクトを返します。
 43  * @param {String} path ファイルのパスを文字列で指定します。
 44  * @param {String} mode ファイルを読み書きするためのモードを指定します。
 45  * @param {Function} block ブロック
 46  * <a href = "File.html#constructor">コンストラクタ</a>参照
 47  */
 48 File.open = function(path, mode, block) {
 49   var file = new File(path, mode);
 50   if(block) {
 51     try {
 52       block(file);
 53     } finally {
 54       file.close();
 55     }
 56   } else {
 57     return file;
 58   }
 59 };

 61 /** 
 62  * fromファイルの名前をtoに変更します。
 63  * @param {String} from 変更前のファイル名
 64  * @param {String} to 変更後のファイル名
 65  * @throws ファイルが存在しない場合にスローされます。
 66  */
 67 File.rename = function(from, to) {
 68   if(Const.FSO.FolderExists(from)) {
 69     Const.FSO.MoveFolder(from, to);
 70   } else if(Const.FSO.FileExists(from)) {
 71     Const.FSO.MoveFile(from, to);
 72   } else {
 73     throw new Error(-1, "File Not Found: " + from);
 74   }
 75 };

 77 /** 
 78  * fromファイルをtoにコピーします。
 79  * @param {String} from コピー元ファイル名
 80  * @param {String} to コピー先ファイル名
 81  * @throws ファイルが存在しない場合にスローされます。
 82  */
 83 File.copy = function(from, to) {
 84   if(Const.FSO.FolderExists(from)) {
 85     Const.FSO.CopyFolder(from, to);
 86   } else if(Const.FSO.FileExists(from)) {
 87     Const.FSO.CopyFile(from, to);
 88   } else {
 89     throw new Error(-1, "File Not Found: " + from);
 90   }
 91 };

 93 /** 
 94  * ファイル、または、ディレクトリが存在するかどうか判定します。
 95  * @param {String} path ファイル、または、ディレクトリのパス
 96  * @return {Boolean} ファイル、または、ディレクトリが存在する場合はtrue、そうでない場合はfalseを返します。
 97  */
 98 File.exist = function(path) {
 99   if(Const.FSO.FolderExists(path)) {
100     return true;
101   } else if(Const.FSO.FileExists(path)) {
102     return true;
103   } else {
104     return false;
105   }
106 };

108 /** 
109  * 指定したpathのファイル、または、ディレクトリを削除します。
110  * @param {String} path ファイル、または、ディレクトリのパス
111  * @param {Boolean} force 読み取り専用も削除するかどうか
112  * @throws ファイルが存在しない場合にスローされます。
113  */
114 File.unlink = function(path, force) {
115   if(!force) {
116     force = false;
117   }
118   if(Const.FSO.FolderExists(path)) {
119     Const.FSO.DeleteFolder(path, force);
120   } else if(Const.FSO.FileExists(path)) {
121     Const.FSO.DeleteFile(path, force);
122   } else {
123     throw new Error(-1, "File Not Found: " + path);
124   }
125 };

127 /** 
128  * 指定したpathのファイル、または、ディレクトリのサイズを取得します。
129  * @param {String} path ファイル、または、ディレクトリのパス
130  * @return {Number} ファイル、または、ディレクトリのサイズ(Byte単位)
131  * @throws ファイルが存在しない場合にスローされます。
132  */
133 File.size = function(path) {
134   if(Const.FSO.FolderExists(path)) {
135     return Const.FSO.GetFolder(path).Size;
136   } else if(Const.FSO.FileExists(path)) {
137     return Const.FSO.GetFile(path).Size;
138   } else {
139     throw new Error(-1, "File Not Found: " + path);
140   }
141 };

143 /** 
144  * 指定したbasedirからの相対パスで、pathnameのフルパスを取得します。
145  * @param {String} pathname ファイル、または、ディレクトリのパス
146  * @param {String} basedir (オプション)起点となるディレクトリ(初期値 カレントディレクトリ)
147  * @return {String} ファイル、または、ディレクトリのフルパス
148  */
149 File.realpath = function(pathname, basedir) {
150   if(!basedir) {
151     basedir = Dir.getwd();
152   }
153 
154   Dir.cd(basedir, function(path) {
155     realpath = Const.FSO.GetAbsolutePathName(pathname);
156   });
157 
158   return realpath;
159 };

161 /** 
162  * 指定したfilenameの拡張子を取得します。
163  * @param {String} filename ファイルのパス
164  * @return {String} filenameの拡張子(ドット付き; .ext)
165  * @example 使用例：
166 print(File.extname("c:\work")); // ""
167 print(File.extname("c:\work\build_y2.xml")); // ".xml"
168  */
169 File.extname = function(filename) {
170   var extensionName = Const.FSO.GetExtensionName(filename);
171   if(extensionName === "") {
172     return extensionName;
173   } else {
174     return "." + extensionName;
175   }
176 };

178 /** 
179  * 指定したfilenameのパスを除いたファイル名を取得します。
180  * @param {String} filename ファイルのパス
181  * @return {String} パスを除いたファイル名(/path/to/file.txt → file.txt)
182  */
183 File.basename = function(filename) {
184   var file = Const.FSO.GetFile(filename);
185   return file.Name;
186 };

188 /** 
189  * 一時ファイルを作成します。(ファイル名:MyDocuments/YYYYMMDDHH24MISSt)
190  * @param {Function} block ブロック
191  */
192 File.createTemporaryFile = function(block) {
193   var tempFilePath = SpecialFolders.getMyDocuments() + "\\" + new Date().getYYYYMMDDHH24MISS();
194   File.open(tempFilePath, "w", function(file) {
195     block(file);
196   });
197 
198   // ファイルが存在していれば削除する。
199   if(File.exist(tempFilePath)) {
200     File.unlink(tempFilePath);
201   }
202 };

204 // Prototypes of File
205 File.prototype = {
206   /** 
207    * ファイルから一行ずつ読み取り、ブロックを呼び出します。
208    * @param {Function} block ブロック
209    */
210   each: function(block) {
211     while(!this.ts.AtEndOfStream) {
212       block(this.ts.ReadLine());
213     }
214   },
215   /** 
216    * ファイルへ一行書き込みます。
217    * @param {String} line 書きこむ文字列
218    */
219   puts: function(line) {
220     this.ts.WriteLine(line);
221   },
222   /** 
223    * ファイルをクローズします。
224    */
225   close: function() {
226     this.ts.Close();
227   },
228   /**
229    * このファイルのパスを取得します。
230    * @return {String} このファイルのパス
231    */
232   getPath: function() {
233     return this.path;
234   }
235 };
