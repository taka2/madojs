 22 /**
 23  * 指定したfnがtrueを返すまで、msec(ミリ秒)間隔でfnを実行します。
 24  * @param {Number} msec スリープするミリ秒
 25  * @param {Function} fn 実行する関数
 26  * @param {Array} args 関数に渡す引数
 27  */
 28 var sleepif = function(msec, fn, args) {
 29   while(true) {
 30     if(fn(args)) {
 31       return true;
 32     }
 33     sleep(msec);
 34   }
 35 };

 79 /**
 80  * 指定したkeyのキー入力を行います。
 81  * @param {String} key 入力する文字列
 82  * @param {Number} num (オプション)送信する回数(初期値 1)
 83  * @param {Number} duration (オプション)送信までの待ち時間(ミリ秒)(初期値 0)
 84  */
 85 var sendKeys = function (key, num, duration) {
 86   var myNum = num || 1;
 87   var myDuration = duration || 0;
 88   for(var i=0; i<myNum; i++) {
 89     sleep(myDuration);
 90     Const.WSHELL.Sendkeys(key);
 91   }
 92   return this;
 93 };

104 /**
105  * OSをシャットダウンします。
106  * @param {Number} timeout (オプション)タイムアウトを秒で指定(初期値 0)
107  */
108 var osShutdown = function (timeout) {
109   var myTimeout = timeout || 0;
110   Process.exec("shutdown", ["-s", "-t", myTimeout], 0, false);
111 };

113 /**
114  * OSを再起動します。
115  * @param {Number} timeout (オプション)タイムアウトを秒で指定(初期値 0)
116  * @param {String} option "-f"を指定すると、実行中のプロセスを警告なしで閉じます
117  */
118 var osReboot = function (timeout, option) {
119   var params = ["-r", "-t"];
120   var myTimeout = timeout || 0;
121   params.push(myTimeout);
122 
123   if (option !== null) {
124     params.push(option);
125   }
126 
127   Process.exec("shutdown", params, 0, false);
128 };

147 /**
148  * ポップアップメッセージを表示します。
149  * @param {String} strText 表示するメッセージ
150  * @param {Number} nSecondsToWait (オプション)ポップアップウインドウを閉じるまでの秒数(初期値 0)
151  * @param {String} strTitle (オプション)ポップアップウインドウのタイトル(初期値 "Windows Script Host")
152  * @param {Number} buttonType (オプション)ポップアップウインドウのボタンの種類(初期値 Const.BUTTON_TYPE_OK)
153  * @param {Number} iconType (オプション)ポップアップウインドウのアイコンの種類(初期値 Const.ICON_TYPE_INFO)
154  * @return {Number} 選択されたボタン値
155  * @example 使用例：
156 if(popup("mado.jsを使いますか？"
157    , 0
158    , "title"
159    , Const.BUTTON_TYPE_YES_NO) === Const.BUTTON_VALUE_YES) {
160    print("はい！");
161 }
162  */
163 var popup = function(strText, nSecondsToWait, strTitle, buttonType, iconType) {
164   var myNSecondsToWait = nSecondsToWait || 0;
165   var myStrTitle = strTitle || "Windows Script Host";
166   var myButtonType = buttonType || Const.BUTTON_TYPE_OK_ONLY;
167   var myIconType = iconType || Const.ICON_TYPE_INFO;
168   return Const.WSHELL.popup(strText, myNSecondsToWait, myStrTitle, myButtonType + myIconType);
169 };

195 /**
196  * JScriptの一次元配列を、一次元のSafeArrayに変換する。<br/>
197  * http://www.imasy.or.jp/~hir/hir/tech/js_tips.html#safearray を参考にしました。
198  * @param {Array} jsArray JScriptの配列を指定します。
199  * @return {Object} 変換されたSafeArrayを返します。通常、VBArrayでラップして使います。
200  */
201 var arrayToSafeArray = function(jsArray) {
202   var dic = new ActiveXObject("Scripting.Dictionary");
203   var jsArrayLength = jsArray.length;
204   for (var i = 0; i < jsArrayLength; i++) {
205     dic.add(i, jsArray[i]);
206   }
207 
208   return dic.items();
209 }
