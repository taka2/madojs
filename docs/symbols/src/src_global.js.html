 11 /**
 12  * 指定したmsec(ミリ秒)スリープします。
 13  * @param {Number} msec スリープするミリ秒
 14  */
 15 var sleep = function(msec) {
 16   WScript.Sleep(msec);
 17 };
 18 
 19 /**
 20  * 指定したfnがtrueを返すまで、msec(ミリ秒)間隔でfnを実行します。
 21  * @param {Number} msec スリープするミリ秒
 22  * @param {Function} fn 実行する関数
 23  * @param {Array} args 関数に渡す引数
 24  */
 25 var sleepif = function(msec, fn, args) {
 26   while(true) {
 27     if(fn(args)) {
 28       return true;
 29     }
 30     sleep(msec);
 31   }
 32 };

 98 /**
 99  * OSをシャットダウンします。
100  * @param {Number} timeout タイムアウトを秒で指定
101  */
102 var osShutdown = function (timeout) {
103   if (timeout === null) timeout = 0;
104   Process.exec("shutdown", ["-s", "-t", timeout], 0, false);
105 };
106 
107 /**
108  * OSを再起動します。
109  * @param {Number} timeout タイムアウトを秒で指定
110  * @param {String} option "-f"を指定すると、実行中のプロセスを警告なしで閉じます
111  */
112 var osReboot = function (timeout, option) {
113   var params = ["-r", "-t"];
114   if (timeout === null) timeout = 0;
115   params.push(timeout);
116 
117   if (option !== null) {
118     params.push(option);
119   }
120 
121   Process.exec("shutdown", params, 0, false);
122 };

126  * @param {String} pathTo 作成するショートカットの場所
127  * @param {String} pathFrom ショートカットの対象となるファイル、または、URL
128  * @example 使用例：

141 /**
142  * ポップアップメッセージを表示します。
143  * @param {String} strText 表示するメッセージ
144  * @param {Number} nSecondsToWait (オプション)ポップアップウインドウを閉じるまでの秒数(初期値 0)
145  * @param {String} strTitle (オプション)ポップアップウインドウのタイトル(初期値 "Windows Script Host")
146  * @param {Number} buttonType (オプション)ポップアップウインドウのボタンの種類(初期値 Const.BUTTON_TYPE_OK)
147  * @param {Number} iconType (オプション)ポップアップウインドウのアイコンの種類(初期値 Const.ICON_TYPE_INFO)
148  * @return {Number} 選択されたボタン値
149  * @example 使用例：
150 if(popup("mado.jsを使いますか？"
151    , 0
152    , "title"
153    , Const.BUTTON_TYPE_YES_NO) === Const.BUTTON_VALUE_YES) {
154    print("はい！");
155 }
156  */
157 var popup = function(strText, nSecondsToWait, strTitle, buttonType, iconType) {
158   var myNSecondsToWait = nSecondsToWait || 0;
159   var myStrTitle = strTitle || "Windows Script Host";
160   var myButtonType = buttonType || Const.BUTTON_TYPE_OK_ONLY;
161   var myIconType = iconType || Const.ICON_TYPE_INFO;
162   return Const.WSHELL.popup(strText, myNSecondsToWait, myStrTitle, myButtonType + myIconType);
163 };
