1 /** 
  2  * 新しいクリップボードを作成する
  3  * @class クリップボードへの格納、および、取得を行うクラス。
  4  */
  5 var Clipboard = function() {
  6   this._ie = new ActiveXObject('InternetExplorer.Application');
  7   this._ie.navigate("about:blank");
  8   while(this._ie.Busy) {
  9     sleep(10);
 10   }
 11   this._ie.Visible = false;
 12   this._textarea = this._ie.document.createElement("textarea");
 13   this._ie.document.body.appendChild(this._textarea);
 14   this._textarea.focus();
 15   this._closed = false;
 16 };
 17 
 18 /** 
 19  * 新しいクリップボードを作成し、ブロックを実行します。
 20  * ブロックが指定されていない場合は、作成したクリップボードを返します。
 21  * @param {Function} block ブロック
 22  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 23  */
 24 Clipboard.open = function(block) {
 25   if (!isFunction(block)) {
 26     return new Clipboard();
 27   }
 28 
 29   try {
 30     var clip = new Clipboard();
 31     block(clip);
 32   } finally {
 33     if (clip != null) {
 34       clip.close();
 35     }
 36   }
 37 };
 38 
 39 // Prototypes of Clipboard
 40 Clipboard.prototype = {
 41   /**
 42    * クリップボードへ文字列を格納します。
 43    * @param {String} text 格納する文字列
 44    */
 45   set: function(text) {
 46     if (this._closed) {
 47       return;
 48     }
 49     this._textarea.innerText = text;
 50     this._ie.execWB(17 /* select all */, 0);
 51     this._ie.execWB(12 /* copy */, 0);
 52   },
 53   /**
 54    * クリップボードの文字列を取得します。
 55    * @return {String} 取得した文字列
 56    */
 57   get: function() {
 58     if (this._closed) {
 59       return;
 60     }
 61     this._textarea.innerText = "";
 62     this._ie.execWB(13 /* paste */, 0);
 63     return this._textarea.innerText;
 64   },
 65   /**
 66    * クリップボードをクローズします。
 67    */
 68   close: function() {
 69     this._ie.Quit();
 70     this._closed = true;
 71   }
 72 };
