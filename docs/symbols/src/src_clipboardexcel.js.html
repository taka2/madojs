1 /** 
  2  * 新しいクリップボードを作成する
  3  * @class Excelを利用した、クリップボードへの格納、および、取得を行うクラス。
  4  */
  5 var ClipboardExcel = function() {
  6   this.excel = Excel.create();
  7   this.sheet = this.excel.getSheetByIndex(0);
  8   this.sheet.setFormat(1, 1, "@");
  9 };
 10 
 11 // Prototypes of ClipboardExcel
 12 ClipboardExcel.prototype = {
 13   /**
 14    * クリップボードへ文字列を格納します。
 15    * @param {String} text 格納する文字列
 16    */
 17   set: function(text) {
 18     this.sheet.setValue(1, 1, text);
 19     this.sheet.copy(1, 1);
 20   },
 21   /**
 22    * クリップボードの文字列を取得します。
 23    * @return {String} 取得した文字列
 24    */
 25   get: function() {
 26     return this.sheet.getValue(1, 1);
 27   },
 28   /**
 29    * クリップボードをクローズします。
 30    */
 31   close: function() {
 32     this.excel.quitDiscardChanges();
 33   }
 34 };
