 10 var sleepif = function(msec, fn, args) {
 11   while(true) {
 12     if(fn(args)) {
 13       return true;
 14     }
 15     sleep(msec);
 16   }
 17 };

 37 var extend = function (destination, source) {

 65 var osShutdown = function (timeout /* タイムアウトをxx秒で指定 */) {
 66   if (timeout === null) timeout = 0;
 67   Process.exec("shutdown", ["-s", "-t", timeout], 0, false);
 68 };
 69 
 70 var osReboot = function (timeout /* タイムアウトをxx秒で指定 */, option /* -f(実行中のプロセスを警告なしで閉じる) */) {
 71   var params = ["-r", "-t"];
 72   if (timeout === null) timeout = 0;
 73   params.push(timeout);
 74 
 75   if (option !== null) {
 76     params.push(option);
 77   }
 78 
 79   Process.exec("shutdown", params, 0, false);
 80 };
