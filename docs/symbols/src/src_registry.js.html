1 /**
  2  * インスタンス化しません。
  3  * @class レジストリにアクセスする機能を提供するクラス
  4 <pre class = "code">
  5 使用例：
  6 // IP Messengerのユーザ名設定を取得
  7 print(Registry.regRead("HKCU\\Software\\HSTools\\IPMsg\\NickNameStr"));
  8 
  9 // "NickNameStra"に"hoge"を設定
 10 Registry.regWrite("HKCU\\Software\\HSTools\\IPMsg\\NickNameStra", "hoge", Registry.REG_TYPE_REG_SZ);
 11 
 12 // "NickNameStra"を削除
 13 Registry.regDelete("HKCU\\Software\\HSTools\\IPMsg\\NickNameStra");
 14 </pre>
 15  */
 16 var Registry = {};
 17 
 18 /**
 19  * レジストリのデータ型：REG_SZ
 20  */
 21 Registry.REG_TYPE_REG_SZ = "REG_SZ";
 22 
 23 /**
 24  * レジストリのデータ型：REG_DWORD
 25  */
 26 Registry.REG_TYPE_REG_DWORD = "REG_DWORD";
 27 
 28 /**
 29  * レジストリのデータ型：REG_BINARY
 30  */
 31 Registry.REG_TYPE_REG_BINARY = "REG_BINARY";
 32 
 33 /**
 34  * レジストリのデータ型：REG_EXPAND_SZ
 35  */
 36 Registry.REG_TYPE_REG_EXPAND_SZ = "REG_EXPAND_SZ";
 37 
 38 /**
 39  * レジストリ内のキー名または値名の値を返します。
 40  * @param {String} name 読み取るキーまたは値の名前です。
 41  * @return {Object} レジストリの値
 42  */
 43 Registry.regRead = function(name) {
 44   return Const.WSHELL.RegRead(name);
 45 };
 46 
 47 /**
 48  * 新しいキーの作成、新しい値名の既存キーへの追加 (および値の設定)、既存の値名の値変更などを行います。
 49  * @param {String} name 作成、追加、変更するキー名、値名、または値を示す文字列値です。
 50  * @param {String} value 作成するキーの名前、既存のキーに追加する値の名前、または既存の値名に設定する値です。
 51  * @param {String} type (オプション)レジストリに保存する値のデータ型を指定します。
 52  */
 53 Registry.regWrite = function(name, value, type) {
 54   var regType = type || Registry.REG_TYPE_REG_SZ;
 55   Const.WSHELL.RegWrite(name, value, regType);
 56 };
 57 
 58 /**
 59  * レジストリから指定されたキーまたは値を削除します。
 60  * @param {String} name レジストリ内で削除するキーまたは値の名前を示す文字列値です。
 61  */
 62 Registry.regDelete = function(name) {
 63   Const.WSHELL.RegDelete(name);
 64 };
