1 /** 
  2  * hostとportを指定してHTTPオブジェクトを作成します。
  3  * @class HTTPの操作を行うためのクラスです。
  4 <pre class = "code">
  5 使用例：
  6 HTTP.start("www.yahoo.co.jp", 80, function(http) {
  7   print(http.get("/index.html"));
  8 });
  9 </pre>
 10  * @param {String} host HTTP操作の対象ホスト
 11  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 12  */
 13 var HTTP = function(host, port) {
 14   this.host = host;
 15   this.port = port || DEFAULT_PORT_NUMBER;
 16 };
 17 
 18 /**
 19  * 定数：デフォルトのポート番号(80)
 20  */
 21 HTTP.DEFAULT_PORT_NUMBER = 80;
 22 /**
 23  * 定数：レスポンス形式（テキスト）
 24  */
 25 HTTP.RESPONSE_TYPE_TEXT = "TEXT";
 26 /**
 27  * 定数：レスポンス形式（ボディ）
 28  */
 29 HTTP.RESPONSE_TYPE_BODY = "BODY";
 30 /**
 31  * 定数：レスポンス形式（XML）
 32  */
 33 HTTP.RESPONSE_TYPE_XML = "XML";
 34 
 35 /** 
 36  * hostとportを指定してHTTPオブジェクトを作成し、ブロックを実行します。
 37  * ブロックが指定されていない場合は、作成したHTTPオブジェクトを返します。
 38  * @param {String} host HTTP操作の対象ホスト
 39  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 40  * @param {Function} block (オプション)ブロック
 41  * @return {Object} ブロックが指定されていない場合は、作成したHTTPオブジェクト
 42  */
 43 HTTP.start = function(host, port, block) {
 44   var http = new HTTP(host, port);
 45   if(block) {
 46     block(http);
 47   } else {
 48     return http;
 49   }
 50 };
 51 
 52 /** 
 53  * 指定されたaddress、path、portに対して、GETリクエストを行います。
 54  * @param {String} address HTTP操作の対象ホスト
 55  * @param {String} path HTTP操作の対象パス
 56  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 57  * @param {Object} header (オプション)HTTPヘッダ(初期値 {})
 58  * @param {String} responseType (オプション)レスポンス形式(初期値 TEXT)
 59  * @return {String} レスポンス
 60  * @example 使用例：
 61 print(HTTP.get("www.yahoo.co.jp", "/index.html", 80));
 62  */
 63 HTTP.get = function(address, path, port, header, responseType) {
 64   var myPort = port || DEFAULT_PORT_NUMBER;
 65   var myHeader = header || {};
 66   var myResponseType = responseType || HTTP.RESPONSE_TYPE_TEXT;
 67   var http = new HTTP(address, myPort);
 68   return http.get(path, myHeader, myResponseType);
 69 };
 70 
 71 /** 
 72  * 指定されたaddress、path、portに対して、GETリクエストを行い、printします。
 73  * @param {String} address HTTP操作の対象ホスト
 74  * @param {String} path HTTP操作の対象パス
 75  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 76  * @param {Object} header (オプション)HTTPヘッダ(初期値 {})
 77  * @param {String} responseType (オプション)レスポンス形式(初期値 TEXT)
 78  * @example 使用例：
 79 HTTP.get_print("www.yahoo.co.jp", "/index.html", 80);
 80  */
 81 HTTP.get_print = function(address, path, port, header, responseType) {
 82   print(HTTP.get(address, path, port, header, responseType));
 83 };
 84 
 85 // Prototypes of HTTP
 86 HTTP.prototype = {
 87   /** 
 88    * 指定されたpath、headerでGETリクエストを行い、ブロックを実行します。
 89    * ブロックが指定されていない場合は、レスポンステキストを返します。
 90    * @param {String} path HTTP操作の対象パス
 91    * @param {Object} header (オプション)HTTPヘッダ(初期値 {})
 92    * @param {String} responseType (オプション)レスポンス形式(初期値 TEXT)
 93    * @param {Function} block (オプション)ブロック
 94    * @return {String} ブロックが指定されていない場合は、レスポンステキスト
 95    * @throws 200OK以外のレスポンスが返された場合にスローされます。
 96    */
 97   get: function(path, header, responseType, block) {
 98     return this.request("GET", path, header, responseType, "", block);
 99   },
100   /** 
101    * 指定されたpath、headerでPOSTリクエストを行い、ブロックを実行します。
102    * ブロックが指定されていない場合は、レスポンステキストを返します。
103    * @param {String} path HTTP操作の対象パス
104    * @param {Object} header (オプション)HTTPヘッダ(初期値 [])
105    * @param {String} responseType (オプション)レスポンス形式(初期値 TEXT)
106    * @param {Object} body (オプション)HTTPボディ(初期値 undefined)
107    * @param {Function} block (オプション)ブロック
108    * @return {String} ブロックが指定されていない場合は、レスポンステキスト
109    * @throws 200OK以外のレスポンスが返された場合にスローされます。
110    */
111   post: function(path, header, responseType, body, block) {
112     return this.request("POST", path, header, responseType, body, block);
113   },
114   // private
115   request: function(method, path, header, responseType, body, block) {
116     // パラメータの調整
117     var myHeader = header || {};
118 
119     // リクエストの準備
120     var xhr = new ActiveXObject("Microsoft.XMLHTTP");
121     xhr.open(method, "http://" + this.host + ":" + this.port + path, false);
122 
123     for(var headerName in myHeader) {
124       if(myHeader.hasOwnProperty(headerName)) {
125         xhr.setRequestHeader(headerName, myHeader[headerName]);
126       }
127     }
128 
129     // リクエストの送信
130     xhr.send(body);
131 
132     // レスポンスの処理
133     if(xhr.readyState === 4) {
134       if(xhr.status === 200) {
135         var response;
136         if(responseType === HTTP.RESPONSE_TYPE_TEXT) {
137           response = xhr.responseText;
138         } else if(responseType === HTTP.RESPONSE_TYPE_BODY) {
139           response = xhr.responseBody;
140         } else if(responseType === HTTP.RESPONSE_TYPE_XML) {
141           response = xhr.responseXML;
142         } else {
143           response = xhr.responseText;
144         }
145         if(isFunction(block)) {
146           block(response);
147         } else {
148           return response;
149         }
150       } else {
151         throw new Error(xhr.statusText);
152       }
153     }
154   }
155 };
