1 /** 
  2  * 新しいキー送信クラスを作成する。
  3  * @class 連続で自動的にキー送信を行う
  4 <pre class = "code">
  5 使用例：
  6 // ノートパッドを起動
  7 Process.exec("notepad", [], Const.WINDOW_STYLE_NORMAL, false);
  8 
  9 // 起動を待つ
 10 sleep(100);
 11 
 12 // "日本語"という文字列を書きこんで、abc.txtに保存。
 13 var ks = new KeySender("無題")
 14     .sendJapaneseKeys("日本語")
 15     .sendKeyWithControl("s")
 16     .sendKeys("abc.txt")
 17     .sendEnter();
 18 </pre>
 19  */
 20 var KeySender = function(targetWindow) {
 21   Const.WSHELL.AppActivate(targetWindow);
 22 };
 23 
 24 // Prototypes of Clipboard
 25 KeySender.prototype = {
 26   /**
 27    * 指定したtextをキー送信します。
 28    * @param {String} text 送信する文字列
 29    * @return this
 30    */
 31   sendKeys: function(text) {
 32     sendKeys(text);
 33     return this;
 34   },
 35   /**
 36    * 指定した日本語textをキー送信します。
 37    * @param {String} text 送信する文字列
 38    * @return this
 39    */
 40   sendJapaneseKeys: function(text) {
 41     Clipboard.open(function(clip) {
 42       clip.set(text);
 43       sendKeys("^v");
 44     });
 45     return this;
 46   },
 47   /**
 48    * タブ文字をキー送信します。
 49    * @return this
 50    */
 51   sendTab: function() {
 52     sendKeys("{TAB}");
 53     return this;
 54   },
 55   /**
 56    * エンターをキー送信します。
 57    * @return this
 58    */
 59   sendEnter: function() {
 60     sendKeys("{ENTER}");
 61     return this;
 62   },
 63   /**
 64    * Shiftを押しながら指定したchをキー送信します。
 65    * @param {String} ch 送信する文字
 66    * @return this
 67    */
 68   sendKeyWithShift: function(ch) {
 69     sendKeys("+" + ch);
 70     return this;
 71   },
 72   /**
 73    * Controlを押しながら指定したchをキー送信します。
 74    * @param {String} ch 送信する文字
 75    * @return this
 76    */
 77   sendKeyWithControl: function(ch) {
 78     sendKeys("^" + ch);
 79     return this;
 80   },
 81   /**
 82    * Altを押しながら指定したchをキー送信します。
 83    * @param {String} ch 送信する文字
 84    * @return this
 85    */
 86   sendKeyWithAlt: function(ch) {
 87     sendKeys("%" + ch);
 88     return this;
 89   }
 90 };
