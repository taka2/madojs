1 /**
  2  * インスタンス化しません。
  3  * @class イベントエントリをログファイルに追加する機能を提供するクラス
  4 <pre class = "code">
  5 使用例：
  6 LogEvent.success("hoge1");
  7 LogEvent.error("hoge2");
  8 LogEvent.warning("hoge3");
  9 LogEvent.information("hoge4");

 12 </pre>

 14 var LogEvent = {};

 16 /**
 17  * イベントエントリをログファイルにSUCCESSで追加します。
 18  * @param {String} strMessage ログエントリのテキストです。 
 19  */
 20 LogEvent.success = function(strMessage) {
 21   Const.WSHELL.LogEvent(Const.LOG_EVENT_TYPE_SUCCESS, strMessage);
 22 };
