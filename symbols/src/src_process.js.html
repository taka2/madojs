1 /**
  2  * インスタンス化しません。
  3  * @class プログラムの実行に関する機能を提供するクラス
  4 <pre class = "code">
  5 使用例：
  6 // ノートパッドを起動
  7 Process.exec("notepad", ["mado-debug.js"], Const.WINDOW_STYLE_NORMAL, false);
  8 </pre>
  9  */
 10 var Process = {};
 11 
 12 /**
 13  * 指定したコマンドを実行します。
 14  * @param {String} command 実行するコマンド
 15  * @param {Array} args (オプション)引数(初期値 [])
 16  * @param {Number} windowStyle (オプション)<a href = "Const.html#.WINDOW_STYLE_MAX">ウインドウスタイル</a>参照(初期値 Const.WINDOW_STYLE_NORMAL)
 17  * @param {Boolean} waitOnReturn (オプション)trueの場合コマンドの実行が終わるまで待ちます。falseの場合は待ちません。(初期値 false)
 18  */
 19 Process.exec = function(command, args, windowStyle, waitOnReturn) {
 20   var myArgs = args || [];
 21   var myWindowStyle = windowStyle || Const.WINDOW_STYLE_NORMAL;
 22   var myWaitOnReturn = waitOnReturn || false;
 23 
 24   var commandLine = command + " " + myArgs.join(" ");
 25   Const.WSHELL.Run(commandLine, myWindowStyle, myWaitOnReturn);
 26 };
