1 /** 
  2  * 新しいADOのコネクションを作成する
  3  * @class ADOを使ったデータベースへの接続を行うクラス。<br/>
  4           通常はこのクラスを直接使用せず、サブクラスを使います。
  5  * @param {String} connectString 接続文字列
  6  * @param {String} userName ユーザ名
  7  * @param {String} password パスワード
  8  */
  9 var AdoConnection = function(connectString, userName, password) {
 10   this.con = new ActiveXObject("ADODB.Connection");
 11   this.con.Open(connectString, userName, password);
 12 };
 13 
 14 /**
 15  * Schema Enumの定数：Table(20)
 16  */
 17 AdoConnection.AD_SCHEMA_TABLES = 20;
 18 
 19 /** 
 20  * 新しいADOのコネクションを作成し、ブロックを実行します。
 21  * ブロックが指定されていない場合は、作成したADOのコネクションを返します。
 22  * @param {String} connectString 接続文字列
 23  * @param {String} userName ユーザ名
 24  * @param {String} password パスワード
 25  * @param {Function} block ブロック
 26  * @return {Object} ブロックが指定されていない場合は、作成したADOのコネクション
 27  */
 28 AdoConnection.open = function(connectString, userName, password, block) {
 29   if (!isFunction(block)) {
 30     return new AdoConnection(connectString, userName, password);
 31   }
 32 
 33   try {
 34     var con = new AdoConnection(connectString, userName, password);
 35     block(con);
 36   } finally {
 37     if(con) {
 38       con.close();
 39     }
 40   }
 41 };
 42 
 43 // Prototypes of AdoConnection
 44 AdoConnection.prototype = {
 45   // private
 46   getRawConnection: function() {
 47     return this.con;
 48   },
 49   /**
 50    * コネクションの接続文字列を取得する。
 51    * @return {String} コネクションの接続文字列
 52    */
 53   getConnectionString: function() {
 54     return this.con.ConnectionString;
 55   },
 56   /**
 57    * 指定されたsqlを実行し、ハッシュ{fieldName: value}の配列として値を返す。
 58    * @param {String} sql 実行するSQL
 59    * @return {Array} ハッシュ{fieldName: value}の配列
 60    */
 61   executeQuery: function(sql) {
 62     try {
 63       // SQLの実行
 64       var rs = this.con.Execute(sql);
 65       return this.convertToArrayRs(rs);
 66     } catch(e) {
 67       throw e;
 68     } finally {
 69       if(rs) {
 70         rs.Close();
 71       }
 72     }
 73   },
 74   /**
 75    * 指定されたsqlを実行する。
 76    * @param {String} sql 実行するSQL
 77    */
 78   executeUpdate: function(sql) {
 79     // SQLの実行
 80     this.con.Execute(sql);
 81   },
 82   // private
 83   // Recordsetを配列にコンバートする。
 84   convertToArrayRs: function(rs) {
 85     // フィールドリストの取得
 86     var fe = new Enumerator(rs.Fields);
 87 
 88     // データの取得
 89     var result = [];
 90 
 91     while(!rs.EOF) {
 92       var record = {};
 93       for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
 94         var field = fe.item();
 95         record[field.Name] = field.Value;
 96       }
 97       result.push(record);
 98 
 99       rs.MoveNext();
100     }
101 
102     return result;
103   },
104   /**
105    * 指定されたsqlを実行するコマンドを作成します。
106    * @param {String} sql 実行するSQL
107    * @return {Object} コマンドオブジェクト
108    */
109   createCommand: function(sql) {
110     return new AdoCommand(this, sql);
111   },
112   /**
113    * データベースの接続を閉じる。
114    */
115   close: function() {
116     this.con.Close();
117   },
118   /**
119    * トランザクションを開始します。
120    * @return {Number} トランザクションネストレベル
121    */
122   beginTrans: function() {
123     return this.con.BeginTrans();
124   },
125   /**
126    * トランザクションをコミットします。
127    */
128   commitTrans: function() {
129     this.con.CommitTrans();
130   },
131   /**
132    * トランザクションをロールバックします。
133    */
134   rollbackTrans: function() {
135     this.con.RollbackTrans();
136   },
137   /**
138    * テーブル名一覧を取得します。
139    * @param {String} schema (オプション)スキーマを指定します。
140    * @return {Array} テーブル名の配列を返します。
141    */
142   getTableNames: function(schema) {
143     var rs = this.con.OpenSchema(AdoConnection.AD_SCHEMA_TABLES, arrayToSafeArray([undefined, schema, undefined, "TABLE"]));
144     var array = this.convertToArrayRs(rs);
145     var result = [];
146     array.each(function(elem) {
147       result.push(elem["TABLE_NAME"]);
148     });
149 
150     return result;
151   },
152   /**
153    * 指定したテーブルのフィールド名一覧を取得します。
154    * @param {String} tableName フィールド名一覧を取得するテーブル名を指定します。
155    * @return {Array} フィールド名一覧を返します。
156    */
157   getFieldNames: function(tableName) {
158     // フィールドリスト取得のためのSQL空実行
159     var rs = this.con.Execute("SELECT * FROM " + tableName);
160 
161     // フィールドリストの取得
162     var fe = new Enumerator(rs.Fields);
163 
164     // データの取得
165     var result = [];
166 
167     for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
168       result.push(fe.item().Name);
169     }
170 
171     return result;
172   }
173 };
