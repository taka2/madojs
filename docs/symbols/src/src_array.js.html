1 /** 
  2  * 各要素に対してブロックを評価します。
  3  * @param {Function} block ブロック
  4  * @example 使用例：
  5  * var x = [1, 2, 3].each(function(i) {
  6  *   return i*2;
  7  * });
  8  *
  9  * print(x); // 2,4,6
 10  */
 11 Array.prototype.each = function(block) {
 12   var result = [];
 13 
 14   var thisLength = this.length;
 15   for(var i=0; i<thisLength; i++) {
 16     result.push(block(this[i]));
 17   }
 18 
 19   return result;
 20 };
 21 
 22 /**
 23  * objが配列に含まれるかどうかチェックします。
 24  * @param {Object} obj 配列に含まれるかどうかチェックするオブジェクト
 25  * @return {Boolean} objが配列に含まれる場合true、そうでない場合falseを返す。
 26  */
 27 Array.prototype.include = function(obj) {
 28   var thisLength = this.length;
 29   for(var i=0; i<thisLength; i++) {
 30     if(this[i] === obj) {
 31       return true;
 32     }
 33   }
 34   return false;
 35 };
