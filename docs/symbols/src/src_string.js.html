1 /** 
  2  * 文字列先頭と末尾のスペースをトリムした文字列を生成して返します。
  3  * @return {String} トリムした文字列
  4  * @example 使用例：
  5 print("@" + "   	abc			".trim() + "@"); // "@abc@"
  6  */
  7 String.prototype.trim = function() {
  8   var text = this.valueOf();
  9   return (text || "").replace( /^\s+|\s+$/g, "" );
 10 };
 11 
 12 /**
 13  * 文字列がchで始まるかどうか
 14  * @param {String} ch テストする文字
 15  * @return {Boolean} 文字列がchで始まる場合true、そうでない場合はfalseを返す。
 16  * @example 使用例：
 17 "abc".startsWith("a"); // true
 18  */
 19 String.prototype.startsWith = function(ch) {
 20   var text = this.valueOf();
 21   return(text.substring(0, 1) === ch);
 22 };
 23 
 24 /**
 25  * 文字列がchで終わるかどうか
 26  * @param {String} ch テストする文字
 27  * @return {Boolean} 文字列がchで終わる場合true、そうでない場合はfalseを返す。
 28  * @example 使用例：
 29 "abc".endsWith("c"); // true
 30  */
 31 String.prototype.endsWith = function(ch) {
 32   var text = this.valueOf();
 33   var textLength = text.length;
 34   return(text.substring(textLength - 1, textLength) === ch);
 35 };
