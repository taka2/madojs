  4           通常はこのクラスを直接使用せず、サブクラスを使います。
  5  * @param {String} connectString 接続文字列
  6  * @param {String} userName ユーザ名

  8  */
  9 var AdoConnection = function(connectString, userName, password) {
 10   this.con = new ActiveXObject("ADODB.Connection");
 11   this.con.Open(connectString, userName, password);
 12 };

180 /** 
181  * 新しいADOのコネクションを作成し、ブロックを実行します。
182  * ブロックが指定されていない場合は、作成したADOのコネクションを返します。
183  * @param {String} connectString 接続文字列
184  * @param {String} userName ユーザ名
185  * @param {String} password パスワード
186  * @param {Function} block ブロック
187  * @return {Object} ブロックが指定されていない場合は、作成したADOのコネクション
188  */
189 AdoConnection.open = function(connectString, userName, password, block) {
190   if (!isFunction(block)) {
191     return new AdoConnection(connectString, userName, password);
192   }
193 
194   try {
195     var con = new AdoConnection(connectString, userName, password);
196     block(con);
197   } finally {
198     if(con) {
199       con.close();
200     }
201   }
202 };

204 // Prototypes of AdoConnection
205 AdoConnection.prototype = {
206   // private
207   getRawConnection: function() {
208     return this.con;
209   },
210   /**
211    * コネクションの接続文字列を取得する。
212    * @return {String} コネクションの接続文字列
213    */
214   getConnectionString: function() {
215     return this.con.ConnectionString;
216   },
217   /**
218    * 指定されたsqlを実行し、ハッシュ{fieldName: value}の配列として値を返す。
219    * @param {String} sql 実行するSQL
220    * @return {Array} ハッシュ{fieldName: value}の配列
221    */
222   executeQuery: function(sql) {
223     try {
224       // SQLの実行
225       var rs = this.con.Execute(sql);
226       return this.convertToArrayRs(rs);
227     } catch(e) {
228       throw e;
229     } finally {
230       if(rs) {
231         rs.Close();
232       }
233     }
234   },
235   /**
236    * 指定されたsqlを実行する。
237    * @param {String} sql 実行するSQL
238    */
239   executeUpdate: function(sql) {
240     // SQLの実行
241     this.con.Execute(sql);
242   },
243   // private
244   // Recordsetを配列にコンバートする。
245   convertToArrayRs: function(rs) {
246     // フィールドリストの取得
247     var fe = new Enumerator(rs.Fields);
248 
249     // データの取得
250     var result = [];
251 
252     while(!rs.EOF) {
253       var record = {};
254       for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
255         var field = fe.item();
256         record[field.Name] = field.Value;
257       }
258       result.push(record);
259 
260       rs.MoveNext();
261     }
262 
263     return result;
264   },
265   /**
266    * 指定されたsqlを実行するコマンドを作成します。
267    * @param {String} sql 実行するSQL
268    * @return {Object} コマンドオブジェクト
269    */
270   createCommand: function(sql) {
271     return new AdoCommand(this, sql);
272   },
273   /**
274    * データベースの接続を閉じる。
275    */
276   close: function() {
277     this.con.Close();
278   },
279   /**
280    * トランザクションを開始します。
281    * @return {Number} トランザクションネストレベル
282    */
283   beginTrans: function() {
284     return this.con.BeginTrans();
285   },
286   /**
287    * トランザクションをコミットします。
288    */
289   commitTrans: function() {
290     this.con.CommitTrans();
291   },
292   /**
293    * トランザクションをロールバックします。
294    */
295   rollbackTrans: function() {
296     this.con.RollbackTrans();
297   },
298   /**
299    * テーブル名一覧を取得します。
300    * @param {String} schema (オプション)スキーマを指定します。
301    * @return {Array} テーブル名の配列を返します。
302    */
303   getTableNames: function(schema) {
304     var rs = this.con.OpenSchema(AdoConnection.AD_SCHEMA_TABLES, arrayToSafeArray([undefined, schema, undefined, "TABLE"]));
305     var array = this.convertToArrayRs(rs);
306     var result = [];
307     array.each(function(elem) {
308       result.push(elem["TABLE_NAME"]);
309     });
310 
311     return result;
312   },
313   /**
314    * 指定したテーブルのフィールド名一覧を取得します。
315    * @param {String} tableName フィールド名一覧を取得するテーブル名を指定します。
316    * @return {Array} フィールド名一覧を返します。
317    */
318   getFieldNames: function(tableName) {
319     // フィールドリスト取得のためのSQL空実行
320     var rs = this.con.Execute("SELECT * FROM " + tableName);
321 
322     // フィールドリストの取得
323     var fe = new Enumerator(rs.Fields);
324 
325     // データの取得
326     var result = [];
327 
328     for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
329       result.push(fe.item().Name);
330     }
331 
332     return result;
333   },
334   /**
335    * 指定したテーブルのフィールド情報一覧を取得します。[{カラム名:カラムタイプ}]
336    * @param {String} tableName フィールド情報一覧を取得するテーブル名を指定します。
337    * @return {Array} フィールド情報一覧を返します。
338    */
339   getFieldInfo: function(tableName) {
340     // フィールドリスト取得のためのSQL空実行
341     var rs = this.con.Execute("SELECT * FROM " + tableName);
342 
343     // フィールドリストの取得
344     var fe = new Enumerator(rs.Fields);
345 
346     // データの取得
347     var result = [];
348 
349     for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
350       var item = fe.item();
351       result.push({'Name': item.Name, 'Type': item.Type});
352     }
353 
354     return result;
355   }
356 };
