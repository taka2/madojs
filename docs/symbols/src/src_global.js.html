 33 /**
 34  * 指定したmsec(ミリ秒)スリープします。<br/>
 35  * hta環境では何もしません。
 36  * @param {Number} msec スリープするミリ秒
 37  */
 38 var sleep = function(msec) {
 39   try {
 40     WScript.Sleep(msec);
 41   } catch(e) {
 42     if(e.name === 'TypeError') {
 43       // 何もしない
 44     } else {
 45       throw e;
 46     }
 47   }
 48 };

 50 /**
 51  * 指定したfnがtrueを返すまで、msec(ミリ秒)間隔でfnを実行します。
 52  * @param {Number} msec スリープするミリ秒
 53  * @param {Function} fn 実行する関数
 54  * @param {Array} args (オプション)関数に渡す引数
 55  */
 56 var sleepif = function(msec, fn, args) {
 57   if(!args) {
 58     args = [];
 59   }
 60   while(true) {
 61     if(fn.apply({}, args)) {
 62       return true;
 63     }
 64 
 65     sleep(msec);
 66   }
 67 };

146 /**
147  * 指定したkeyのキー入力を行います。
148  * @param {String} key 入力する文字列
149  * @param {Number} num (オプション)送信する回数(初期値 1)
150  * @param {Number} duration (オプション)送信までの待ち時間(ミリ秒)(初期値 0)
151  */
152 var sendKeys = function (key, num, duration) {
153   var myNum = num || 1;
154   var myDuration = duration || 0;
155   for(var i=0; i<myNum; i++) {
156     sleep(myDuration);
157     Const.WSHELL.Sendkeys(key);
158   }
159   return this;
160 };

171 /**
172  * OSをシャットダウンします。
173  * @param {Number} timeout (オプション)タイムアウトを秒で指定(初期値 0)
174  */
175 var osShutdown = function (timeout) {
176   var myTimeout = timeout || 0;
177   Process.exec("shutdown", ["-s", "-t", myTimeout], 0, false);
178 };

180 /**
181  * OSを再起動します。
182  * @param {Number} timeout (オプション)タイムアウトを秒で指定(初期値 0)
183  * @param {String} option "-f"を指定すると、実行中のプロセスを警告なしで閉じます
184  */
185 var osReboot = function (timeout, option) {
186   var params = ["-r", "-t"];
187   var myTimeout = timeout || 0;
188   params.push(myTimeout);
189 
190   if (option !== null) {
191     params.push(option);
192   }
193 
194   Process.exec("shutdown", params, 0, false);
195 };

214 /**
215  * ポップアップメッセージを表示します。
216  * @param {String} strText 表示するメッセージ
217  * @param {Number} nSecondsToWait (オプション)ポップアップウインドウを閉じるまでの秒数(初期値 0)
218  * @param {String} strTitle (オプション)ポップアップウインドウのタイトル(初期値 "Windows Script Host")
219  * @param {Number} buttonType (オプション)ポップアップウインドウのボタンの種類(初期値 Const.BUTTON_TYPE_OK)
220  * @param {Number} iconType (オプション)ポップアップウインドウのアイコンの種類(初期値 Const.ICON_TYPE_INFO)
221  * @return {Number} 選択されたボタン値
222  * @example 使用例：
223 if(popup("mado.jsを使いますか？"
224    , 0
225    , "title"
226    , Const.BUTTON_TYPE_YES_NO) === Const.BUTTON_VALUE_YES) {
227    print("はい！");
228 }
229  */
230 var popup = function(strText, nSecondsToWait, strTitle, buttonType, iconType) {
231   var myNSecondsToWait = nSecondsToWait || 0;
232   var myStrTitle = strTitle || "Windows Script Host";
233   var myButtonType = buttonType || Const.BUTTON_TYPE_OK_ONLY;
234   var myIconType = iconType || Const.ICON_TYPE_INFO;
235   return Const.WSHELL.popup(strText, myNSecondsToWait, myStrTitle, myButtonType + myIconType);
236 };

262 /**
263  * JScriptの一次元配列を、一次元のSafeArrayに変換する。<br/>
264  * http://www.imasy.or.jp/~hir/hir/tech/js_tips.html#safearray を参考にしました。
265  * @param {Array} jsArray JScriptの配列を指定します。
266  * @return {Object} 変換されたSafeArrayを返します。通常、VBArrayでラップして使います。
267  */
268 var arrayToSafeArray = function(jsArray) {
269   var dic = new ActiveXObject("Scripting.Dictionary");
270   var jsArrayLength = jsArray.length;
271   for (var i = 0; i < jsArrayLength; i++) {
272     dic.add(i, jsArray[i]);
273   }
274 
275   return dic.items();
276 }
