1 // オリジナルのtoStringをorgToStringとして保存
  2 Array.prototype.orgToString = Array.prototype.toString;
  3 
  4 /**
  5  * オブジェクトの値を表す文字列を返します。
  6  * @return {String} オブジェクトの値を表す文字列
  7  */
  8 Array.prototype.toString = function() {
  9   var result = "[";
 10   var length = this.length;
 11   for(var i=0; i<length; i++) {
 12     if(result !== "[") {
 13       result += ", ";
 14     }
 15     result += this[i].toString();
 16   }
 17   result += "]";
 18   return result;
 19 };

 21 /** 
 22  * 各要素に対してブロックを評価します。
 23  * @param {Function} block ブロック
 24  * @example 使用例：
 25  * var x = [1, 2, 3].forEach(function(i) {
 26  *   return i*2;
 27  * });
 28  *
 29  * print(x); // 2,4,6
 30  */
 31 Array.prototype.forEach = function(block) {
 32   var result = [];
 33 
 34   var thisLength = this.length;
 35   for(var i=0; i<thisLength; i++) {
 36     result.push(block(this[i]));
 37   }
 38 
 39   return result;
 40 };

 42 Array.prototype.each = Array.prototype.forEach;

 44 /**
 45  * objが配列に含まれるかどうかチェックします。
 46  * @param {Object} obj 配列に含まれるかどうかチェックするオブジェクト
 47  * @return {Boolean} objが配列に含まれる場合true、そうでない場合falseを返す。
 48  */
 49 Array.prototype.include = function(obj) {
 50   var thisLength = this.length;
 51   for(var i=0; i<thisLength; i++) {
 52     if(this[i] === obj) {
 53       return true;
 54     }
 55   }
 56   return false;
 57 };
