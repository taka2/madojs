1 /** 
  2  * 新しいADOのコネクションを作成する
  3  * @class ADOを使ったデータベースへの接続を行うクラス
  4  * @param {String} connectString 接続文字列
  5  * @param {String} userName ユーザ名
  6  * @param {String} password パスワード
  7  */
  8 var AdoConnection = function(connectString, userName, password) {
  9   this.con = new ActiveXObject("ADODB.Connection");
 10   this.con.Open(connectString, userName, password);
 11 };
 12 
 13 /** 
 14  * 新しいADOのコネクションを作成し、ブロックを実行します。
 15  * ブロックが指定されていない場合は、作成したADOのコネクションを返します。
 16  * @param {String} connectString 接続文字列
 17  * @param {String} userName ユーザ名
 18  * @param {String} password パスワード
 19  * @param {Function} block ブロック
 20  * @return {Object} ブロックが指定されていない場合は、作成したADOのコネクション
 21  */
 22 AdoConnection.open = function(connectString, userName, password, block) {
 23   if (!isFunction(block)) {
 24     return new AdoConnection(connectString, userName, password);
 25   }
 26 
 27   try {
 28     var con = new AdoConnection(connectString, userName, password);
 29     block(con);
 30   } finally {
 31     if (con != null) {
 32       con.close();
 33     }
 34   }
 35 };
 36 
 37 // Prototypes of AdoConnection
 38 AdoConnection.prototype = {
 39   // private
 40   getRawConnection: function() {
 41     return this.con;
 42   },
 43   /**
 44    * コネクションの接続文字列を取得する。
 45    * @return {String} コネクションの接続文字列
 46    */
 47   getConnectionString: function() {
 48     return this.con.ConnectionString;
 49   },
 50   /**
 51    * 指定されたsqlを実行し、ハッシュ{fieldName: value}の配列として値を返す。
 52    * @param {String} sql 実行するSQL
 53    * @return {Array} ハッシュ{fieldName: value}の配列
 54    */
 55   executeQuery: function(sql) {
 56     try {
 57       // SQLの実行
 58       var rs = this.con.Execute(sql);
 59 
 60       // フィールドリストの取得
 61       var fe = new Enumerator(rs.Fields);
 62 
 63       // データの取得
 64       var result = [];
 65 
 66       while(!rs.EOF) {
 67         var record = {};
 68         for(fe.moveFirst(); !fe.atEnd(); fe.moveNext()) {
 69           var field = fe.item();
 70           record[field.Name] = field.Value;
 71         }
 72         result.push(record);
 73 
 74         rs.MoveNext();
 75       }
 76 
 77       return result;
 78     } catch(e) {
 79       throw e;
 80     } finally {
 81       if(rs) {
 82         rs.Close();
 83       }
 84     }
 85   },
 86   /**
 87    * データベースの接続を閉じる。
 88    */
 89   close: function() {
 90     this.con.Close();
 91   }
 92 };
