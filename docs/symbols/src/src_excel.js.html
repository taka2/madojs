1 /** 
  2  * Excelオブジェクトを作成します。
  3  * @class Excelファイルの操作と読み書きを行うためのクラスです。<br/>
  4  * Excelがインストールされている必要があります。
  5 <pre class = "code">
  6 使用例：
  7 // abc.xlsを開いて、Sheet1の、1行1列目の値を'hoge'に書き換えつつ、上書き保存する。
  8 Excel.open("abc.xls", function(excel) {
  9   var sheet = excel.getSheetByName("Sheet1");
 10   sheet.setValue(1, 1, 'hoge');
 11   excel.save();
 12 });
 13 </pre>
 14  * @param {String} path Excelファイルのパスを文字列で指定します。指定しない場合は新しくブックを作成します。
 15  * @param {Boolean} fileType (オプション)Excel以外のファイルを開くときに、指定します。
 16  * @param {Boolean} columnInfo (オプション)Excel以外のファイルを開くときに、カラムの情報を指定します。
 17  * @throws pathがundefinedでない、かつ、ファイルが存在しない場合にスローされます。
 18  */
 19 var Excel = function(path, fileType, columnInfo) {
 20   this.path = path;
 21   this.excelObj = new ActiveXObject("Excel.Application");
 22 
 23   if(path === undefined) {
 24     this.workbookObj = this.excelObj.Workbooks.Add();
 25   } else {
 26     if(!File.exist(path)) {
 27       throw new Error(-1, "File Not Found: " + path);
 28     }
 29 
 30     switch(fileType) {
 31       case Excel.IN_FILETYPE_CSV:
 32       case Excel.IN_FILETYPE_TSV:
 33         var isFileTypeCsv = fileType === Excel.IN_FILETYPE_CSV;
 34         var isFileTypeTsv = fileType === Excel.IN_FILETYPE_TSV;
 35 
 36         var columnInfoSafeArray = Excel.array2dToSafeArray2d(columnInfo);
 37         this.excelObj.Workbooks.OpenText(path, 1, 1, 1, 1, false, isFileTypeTsv, false, isFileTypeCsv, false, false, false, columnInfoSafeArray);
 38         this.workbookObj = this.excelObj.ActiveWorkbook;
 39         break;
 40       default:
 41         this.workbookObj = this.excelObj.Workbooks.Open(path);
 42     }
 43   }
 44 };

267 /**
268  * JScriptの二次元配列を、二次元のSafeArrayに変換する。
269  * @param {Array} jsArray2d JScriptの二次元配列を指定します。
270  * @return {Object} 変換された二次元のSafeArrayを返します。通常、VBArrayでラップして使います。<br/>
271  *                  変換に失敗した場合はundefinedを返します。
272  */
273 Excel.array2dToSafeArray2d = function(jsArray2d) {
274   var safeArray = undefined;
275 
276   Excel.createReadonly(function(excel) {
277     // 1シート目を取得
278     var sheet = excel.getSheetByIndex(1);
279 
280     // 各セルに値を設定
281     for(i=0; i<jsArray2d.length; i++) {
282       for(j=0; j<jsArray2d[i].length; j++) {
283         var value = jsArray2d[i][j];
284         sheet.setValue(i+1, j+1, value);
285       }
286     }
287 
288     // SafeArrayを取り出す。
289     safeArray = sheet.getRawObject().Cells(1, 1).CurrentRegion.Value;
290   });
291 
292   return safeArray;
293 }

295 /**
296  * CSVファイルをExcelファイルに変換する。
297  * @param {String} path CSVファイルのパスを文字列で指定します。
298  * @param {Array} columnInfo (オプション)CSVファイルのカラム情報を指定します。<br>
299 <a href = "Excel.html#.COLUMN_TYPE_GENERAL">カラムタイプ</a>参照(初期値 全列に対しConst.COLUMN_TYPE_TEXT)
300  * @return {Boolean} 成功した場合はtrue、失敗した場合はfalseを返す。
301  * @example 使用例：
302 // 3カラム全て文字列として取り込む
303 var columnInfo = [
304 	[1, Excel.COLUMN_TYPE_TEXT]
305 	, [2, Excel.COLUMN_TYPE_TEXT]
306 	, [3, Excel.COLUMN_TYPE_TEXT]
307 ];
308 
309 // 拡張子がCSVだと、Excelが自動的に変換してうまくいかない
310 Excel.convertFromCsv("csv.txt", columnInfo);
311  */
312 Excel.convertFromCsv = function(path, columnInfo) {
313   var myPath = path;
314 
315   // 拡張子が.csvの場合は、うまく変換できないので、拡張子に.txtを付与したファイルにコピーする。
316   if(path.endsWith(".csv")) {
317     myPath = path + ".txt";
318     File.copy(path, myPath);
319   }
320 
321   var myColumnInfo = columnInfo;
322   if(columnInfo === undefined) {
323     myColumnInfo = Excel.generateDefaultColumnInfo(myPath, ",");
324   }
325 
326   var excel = new Excel(myPath, Excel.IN_FILETYPE_CSV, myColumnInfo);
327   excel.saveAsExcel(path + ".xls");
328   excel.quit();
329 
330   // 一時ファイルを削除する
331   if(path !== myPath) {
332     File.unlink(myPath);
333   }
334 }

343 Excel.convertFromTsv = function(path, columnInfo) {

347   }

354 // 1行目の内容に基づいてデフォルトのcolumnInfoを生成する。
355 Excel.generateDefaultColumnInfo = function(path, separateChar) {
356   var columnInfo = [];
357 
358   File.open(path, "r", function(file) {
359     var firstLine = file.readLine();
360     var columnCount = firstLine.split(separateChar).length;
361 
362     var i;
363     for(i=0; i<columnCount; i++) {
364       columnInfo.push([(i+1), Excel.COLUMN_TYPE_TEXT]);
365     }
366   });
367 
368   return columnInfo;
369 }

374    * ワークブックに含まれる、ワークシートを処理します。
375    * @param {Function} block ブロック
376    */
377   each: function(block) {
378     var enumSheet = new Enumerator(this.workbookObj.Sheets);
379     for (;!enumSheet.atEnd(); enumSheet.moveNext()) {
380        block(new ExcelSheet(enumSheet.item()));
381     }
382   },
383   /**
384    * 指定したsheetNameのワークシートを取得します。
385    * @param {String} sheetName 取得するワークシート名
386    * @return {Object} ワークシートオブジェクト
387    */
388   getSheetByName: function(sheetName) {
389     var result;
390     this.each(function(sheet) {
391       if(sheet.getName() === sheetName) {
392         result = sheet;
393       }
394     });
395 
396     return result;
397   },
398   /**
399    * 指定したindexのワークシートを取得します。
400    * @param {Number} index 取得するワークシートのインデックス番号
401    * @return {ExcelSheet} ワークシートオブジェクト
402    */
403   getSheetByIndex: function(index) {
404     var result;
405     var i=0;
406     this.each(function(sheet) {
407       if(i === index) {
408         result = sheet;
409       }
410       i++;
411     });
412 
413     return result;
414   },
415   /**
416    * 指定したindexのワークシートの後にシートを1枚追加します。
417    * @param {String} sheetName 追加するシートのシート名

419    * @return {ExcelSheet} ワークシートオブジェクト
420    */
421   addSheet: function(sheetName, index) {
422     var myIndex = index || this.getSheetCount() - 1;
423 
424     var targetSheet = this.getSheetByIndex(myIndex).getRawObject();
425     var addedSheet = this.workbookObj.Sheets.Add(null, targetSheet, 1, null);
426     addedSheet.Name = sheetName;
427 
428     return new ExcelSheet(addedSheet);
429   },
430   /**
431    * ワークシート数を取得します。
432    * @return {Number} ワークシート数
433    */

436   },

442   },

444     var myFileName = fileName;
445     if(myFileName.indexOf(File.PATH_SEPARATOR) === -1) {
446       myFileName = File.realpath(myFileName);
447     }
448

450   },
