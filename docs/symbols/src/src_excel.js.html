1 /** 
  2  * Excelオブジェクトを作成します。
  3  * @class Excelファイルの操作と読み書きを行うためのクラスです。<br/>
  4  * Excelがインストールされている必要があります。
  5 <pre class = "code">
  6 使用例：
  7 // abc.xlsを開いて、Sheet1の、1行1列目の値を'hoge'に書き換えつつ、上書き保存する。
  8 Excel.open("abc.xls", function(excel) {
  9   var sheet = excel.getSheetByName("Sheet1");
 10   sheet.setValue(1, 1, 'hoge');
 11   excel.save();
 12 });
 13 </pre>
 14  * @param {String} path Excelファイルのパスを文字列で指定します。
 15  * @param {Boolean} bNew (オプション)新しくファイルを作成するかどうか(初期値 false)
 16  * @throws bNewがfalse、かつ、ファイルが存在しない場合にスローされます。
 17  */
 18 var Excel = function(path, bNew) {
 19   this.path = path;
 20 
 21   this.excelObj = new ActiveXObject("Excel.Application");
 22   if(bNew) {
 23     this.workbookObj = this.excelObj.Workbooks.Add();
 24   } else {
 25     if(!File.exist(path)) {
 26       throw new Error(-1, "File Not Found: " + path);
 27     }
 28 
 29     this.workbookObj = this.excelObj.Workbooks.Open(path);
 30   }
 31 };

 33 /** 
 34  * Excelファイルを開き、ブロックを実行します。
 35  * ブロックが指定されていない場合は、Excelオブジェクトを返します。
 36  * @param {String} path Excelファイルのパスを文字列で指定します。
 37  * @param {Function} block ブロック
 38  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 39  */
 40 Excel.open = function(path, block) {
 41   if(!isFunction(block)) {
 42     return new Excel(path, false);
 43   }
 44 
 45   try {
 46     var excel = new Excel(path, false);
 47     block(excel);
 48   } finally {
 49     if (excel != null) {
 50       excel.quit();
 51     }
 52   }
 53 };

 55 /** 
 56  * Excelファイルを読み取り専用(最後に変更を破棄)で開き、ブロックを実行します。
 57  * ブロックが指定されていない場合は、Excelオブジェクトを返します。
 58  * @param {String} path Excelファイルのパスを文字列で指定します。
 59  * @param {Function} block ブロック
 60  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 61  */
 62 Excel.openReadonly = function(path, block) {
 63   if(!isFunction(block)) {
 64     return new Excel(path, false);
 65   }
 66 
 67   try {
 68     var excel = new Excel(path, false);
 69     block(excel);
 70   } finally {
 71     if (excel != null) {
 72       excel.quitDiscardChanges();
 73     }
 74   }
 75 };

 77 /** 
 78  * Excelファイルを作成し、ブロックを実行します。
 79  * ブロックが指定されていない場合は、Excelオブジェクトを返します。
 80  * @param {String} path Excelファイルのパスを文字列で指定します。
 81  * @param {Function} block ブロック
 82  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 83  */
 84 Excel.create = function(block) {
 85   if(!isFunction(block)) {
 86     return new Excel(path, true);
 87   }
 88 
 89   try {
 90     var excel = new Excel("", true);
 91     block(excel);
 92   } finally {
 93     if (excel != null) {
 94       excel.quit();
 95     }
 96   }
 97 };

 99 // Prototypes of Excel
100 Excel.prototype = {
101   /**
102    * ワークブックに含まれる、ワークシートを処理します。
103    * @param {Function} block ブロック
104    */
105   each: function(block) {
106     var enumSheet = new Enumerator(this.workbookObj.Sheets);
107     for (;!enumSheet.atEnd(); enumSheet.moveNext()) {
108        block(new ExcelSheet(enumSheet.item()));
109     }
110   },
111   /**
112    * 指定したsheetNameのワークシートを取得します。
113    * @param {String} sheetName 取得するワークシート名
114    * @return {Object} ワークシートオブジェクト
115    */
116   getSheetByName: function(sheetName) {
117     var result;
118     this.each(function(sheet) {
119       if(sheet.getName() === sheetName) {
120         result = sheet;
121       }
122     });
123 
124     return result;
125   },
126   /**
127    * 指定したindexのワークシートを取得します。
128    * @param {Number} index 取得するワークシートのインデックス番号
129    * @return {ExcelSheet} ワークシートオブジェクト
130    */
131   getSheetByIndex: function(index) {
132     var result;
133     var i=0;
134     this.each(function(sheet) {
135       if(i === index) {
136         result = sheet;
137       }
138       i++;
139     });
140 
141     return result;
142   },
143   /**
144    * 指定したindexのワークシートの後にシートを1枚追加します。
145    * @param {String} sheetName 追加するシートのシート名
146    * @param {Number} index (オプション)追加するシートの位置(初期値 最後尾)
147    * @return {ExcelSheet} ワークシートオブジェクト
148    */
149   addSheet: function(sheetName, index) {
150     var myIndex = index || this.getSheetCount() - 1;
151 
152     var targetSheet = this.getSheetByIndex(myIndex).getRawObject();
153     var addedSheet = this.workbookObj.Sheets.Add(null, targetSheet, 1, null);
154     addedSheet.Name = sheetName;
155 
156     return new ExcelSheet(addedSheet);
157   },
158   /**
159    * ワークシート数を取得します。
160    * @return {Number} ワークシート数
161    */
162   getSheetCount: function() {
163     return this.workbookObj.Sheets.Count;
164   },
165   /** 
166    * Excelファイルを保存します。
167    */
168   save: function() {
169     this.workbookObj.Save();
170   },
171   /** 
172    * Excelファイルをファイル名を指定して保存します。
173    */
174   saveAs: function(fileName) {
175     this.workbookObj.SaveAs(fileName);
176   },
177   /**
178    * Excelファイルを保存せずに閉じます
179    */
180   quitDiscardChanges: function() {
181     this.workbookObj.Close(false);
182     this.excelObj.Quit();
183   },
184   /** 
185    * Excelを終了します。<br/>
186    * 未保存の変更がある場合は、ダイアログが表示されます。
187    */
188   quit: function() {
189     this.workbookObj.Close();
190     this.excelObj.Quit();
191   }
192 };

194 /** 
195  * Excelシートオブジェクトを作成します。
196  * @class Excelシートの操作と読み書きを行うためのクラスです。<br/>
197  * Excelがインストールされている必要があります。
198  * @param {String} sheetObj Worksheetオブジェクトを指定します。
199  */
200 var ExcelSheet = function(sheetObj) {
201   this.sheetObj = sheetObj;
202 };
203 
204 // Prototypes of ExcelSheet
205 ExcelSheet.prototype = {
206   /**
207    * ワークシート名を取得します。
208    * @return {String} ワークシート名
209    */
210   getName: function() {
211     return this.sheetObj.Name;
212   },
213   /**
214    * ワークシート名を設定します。
215    * @param {String} sheetName ワークシート名
216    */
217   setName: function(sheetName) {
218     return this.sheetObj.Name = sheetName;
219   },
220   /**
221    * セルに値を設定します。
222    * @param {Number} x x座標
223    * @param {Number} y y座標
224    * @param {Object} value 設定する値
225    */
226   setValue: function(x, y, value) {
227     this.sheetObj.Cells(x, y).Value = value;
228   },
229   /**
230    * セルの値を取得します。
231    * @param {Number} x x座標
232    * @param {Number} y y座標
233    * @return {Object} セルの値
234    */
235   getValue: function(x, y) {
236     return this.sheetObj.Cells(x, y).Value;
237   },
238   /**
239    * ラップしている生のWorksheetオブジェクトを返します。<br/>
240    * 通常使いません。
241    */
242   getRawObject: function() {
243     return this.sheetObj;
244   },
245   /**
246    * オートフィットを行います。
247    */
248   autoFit: function() {
249     this.sheetObj.Columns("A:IV").AutoFit();
250   },
251   /**
252    * カレントリージョンに対して罫線を引きます。
253    */
254   drawBorder: function() {
255     var range = this.sheetObj.Cells(1, 1).CurrentRegion;
256     var toIndex = 12;
257     if(range.Rows.Count === 1) {
258       // 1行しかない場合は12は引かない
259       toIndex = 11;
260     }
261     for(var i=7; i<=toIndex; i++) {
262       range.Borders(i).LineStyle = 1;
263     }
264   },
265   /**
266    * セルの書式を設定します。
267    * @param {Number} x x座標
268    * @param {Number} y y座標
269    * @param {String} format 書式
270    */
271   setFormat: function(x, y, format) {
272     this.sheetObj.Cells(x, y).NumberFormatLocal = format;
273   },
274   /**
275    * セルにコメントを追加します。
276    * @param {Number} x x座標
277    * @param {Number} y y座標
278    * @param {String} comment コメント
279    * @param {Boolean} visible コメントを表示するかどうか
280    */
281   addComment: function(x, y, comment, visible) {
282     this.sheetObj.Cells(x, y).AddComment();
283     this.sheetObj.Cells(x, y).Comment.Visible = visible;
284     this.sheetObj.Cells(x, y).Comment.Text(comment);
285   },
286   /**
287    * シートをアクティブ化します。
288    */
289   activate: function() {
290     this.sheetObj.Activate();
291   },
292   /**
293    * シートを削除します。
294    */
295   remove: function() {
296     this.sheetObj.Delete();
297   },
298   /**
299    * ズームを設定します。
300    * @param {Number} zoom 拡大率(整数)
301    */
302   setZoom: function(zoom) {
303     // シートをアクテイブ化する
304     this.activate();
305 
306     // 拡大率を設定する
307     this.sheetObj.Application.ActiveWindow.Zoom = zoom;
308   },
309   /**
310    * センターヘッダを設定します。
311    * @param {String} header センターヘッダに設定する内容
312    */
313   setCenterHeader: function(header) {
314     this.sheetObj.PageSetup.CenterHeader = header;
315   },
316   /**
317    * レフトヘッダを設定します。
318    * @param {String} header レフトヘッダに設定する内容
319    */
320   setLeftHeader: function(header) {
321     this.sheetObj.PageSetup.LeftHeader = header;
322   },
323   /**
324    * ライトヘッダを設定します。
325    * @param {String} header ライトヘッダに設定する内容
326    */
327   setRightHeader: function(header) {
328     this.sheetObj.PageSetup.RightHeader = header;
329   },
330   /**
331    * センターフッタを設定します。
332    * @param {String} footer センターフッタに設定する内容
333    */
334   setCenterFooter: function(footer) {
335     this.sheetObj.PageSetup.CenterFooter = footer;
336   },
337   /**
338    * レフトフッタを設定します。
339    * @param {String} footer レフトフッタに設定する内容
340    */
341   setLeftFooter: function(footer) {
342     this.sheetObj.PageSetup.LeftFooter = footer;
343   },
344   /**
345    * ライトフッタを設定します。
346    * @param {String} footer ライトフッタに設定する内容
347    */
348   setRightFooter: function(footer) {
349     this.sheetObj.PageSetup.RightFooter = footer;
350   }
