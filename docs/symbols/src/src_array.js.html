1 /** 
  2  * 各要素に対してブロックを評価します。
  3  * @param {Function} block ブロック
  4  * @example 使用例：
  5  * var x = [1, 2, 3].each(function(i) {
  6  *   return i*2;
  7  * });
  8  *
  9  * print(x); // 2,4,6
 10  */
 11 Array.prototype.each = function(block) {
 12   var result = [];
 13 
 14   for(var i=0; i<this.length; i++) {
 15     result.push(block(this[i]));
 16   }
 17 
 18   return result;
 19 };
