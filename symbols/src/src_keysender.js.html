1 /** 
  2  * 新しいキー送信クラスを作成する。
  3  * @class 連続で自動的にキー送信を行う
  4 <pre class = "code">
  5 使用例：
  6 // ノートパッドを起動
  7 Process.exec("notepad", [], Const.WINDOW_STYLE_NORMAL, false);
  8 
  9 // 起動を待つ
 10 sleep(100);
 11 
 12 // "日本語"という文字列を書きこんで、abc.txtに保存。
 13 var ks = new KeySender("無題")
 14     .sendJapaneseKeys("日本語")
 15     .sendKeyWithControl("s")
 16     .sendKeys("abc.txt")
 17     .sendEnter();
 18 </pre>
 19  * @param {String} targetWindow KeySend対象のウインドウ名を指定します。
 20  * @param {Number} duration (オプション)KeySendごとの待ち時間(ミリ秒)を指定します(初期値 0)。
 21  */
 22 var KeySender = function(targetWindow, duration) {
 23   this.myDuration = duration || 0;
 24   Const.WSHELL.AppActivate(targetWindow);
 25   sleep(KeySender.DURATION);
 26 };
 27 
 28 // Constants of KeySender
 29 // private
 30 // AppActivateしてからKeySendするまでの待ち時間
 31 KeySender.DURATION = 10;

 39    * @return this
 40    */
 41   sendKeys: function(text, num) {
 42     sendKeys(text, num, this.myDuration);
 43     return this;
 44   },
 45   /**
 46    * 指定した日本語textをキー送信します。

 48    * @param {Number} num (オプション)送信する回数(初期値 1)
 49    * @return this
 50    */
 51   sendJapaneseKeys: function(text, num) {
 52     var myNum = num || 1;
 53     Clipboard.open(function(clip) {
 54       for(var i=0; i<myNum; i++) {
 55         clip.set(text);
 56         sendKeys("^v", 1, this.myDuration);
 57       }
 58     });
 59     return this;
 60   },
