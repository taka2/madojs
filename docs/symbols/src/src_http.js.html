1 /** 
  2  * hostとportを指定してHTTPオブジェクトを作成します。
  3  * @class HTTPの操作を行うためのクラスです。
  4  * @param {String} host HTTP操作の対象ホスト
  5  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
  6  */
  7 var HTTP = function(host, port) {
  8   this.host = host;
  9   this.port = port || DEFAULT_PORT_NUMBER;
 10 };
 11 
 12 /**
 13  * 定数：デフォルトのポート番号(80)
 14  */
 15 HTTP.DEFAULT_PORT_NUMBER = 80;
 16 
 17 /** 
 18  * hostとportを指定してHTTPオブジェクトを作成し、ブロックを実行します。
 19  * ブロックが指定されていない場合は、作成したHTTPオブジェクトを返します。
 20  * @param {String} host HTTP操作の対象ホスト
 21  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 22  * @param {Function} block (オプション)ブロック
 23  * @return {Object} ブロックが指定されていない場合は、作成したHTTPオブジェクト
 24  * @example 使用例：
 25  * HTTP.start("www.yahoo.co.jp", 80, function(http) {
 26  *   print(http.get("/index.html"));
 27  * });
 28  */
 29 HTTP.start = function(host, port, block) {
 30   var http = new HTTP(host, port);
 31   if(block) {
 32     block(http);
 33   } else {
 34     return http;
 35   }
 36 };
 37 
 38 /** 
 39  * 指定されたaddress、path、portに対して、GETリクエストを行います。
 40  * @param {String} address HTTP操作の対象ホスト
 41  * @param {String} path HTTP操作の対象パス
 42  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 43  * @return {String} レスポンステキスト
 44  */
 45 HTTP.get = function(address, path, port) {
 46   var myPort = port || DEFAULT_PORT_NUMBER;
 47   var http = new HTTP(address, port);
 48   return http.get(path);
 49 };
 50 
 51 /** 
 52  * 指定されたaddress、path、portに対して、GETリクエストを行い、printします。
 53  * @param {String} address HTTP操作の対象ホスト
 54  * @param {String} path HTTP操作の対象パス
 55  * @param {Number} port (オプション)HTTP操作の対象ポート(初期値 80)
 56  */
 57 HTTP.get_print = function(address, path, port) {
 58   print(HTTP.get(address, path, port));
 59 };
 60 
 61 // Prototypes of HTTP
 62 HTTP.prototype = {
 63   /** 
 64    * 指定されたpath、headerでGETリクエストを行い、ブロックを実行します。
 65    * ブロックが指定されていない場合は、レスポンステキストを返します。
 66    * @param {String} path HTTP操作の対象パス
 67    * @param {Object} header (オプション)HTTPヘッダ
 68    * @param {Function} block (オプション)ブロック
 69    * @return {String} ブロックが指定されていない場合は、レスポンステキスト
 70    * @throws 200OK以外のレスポンスが返された場合にスローされます。
 71    */
 72   get: function(path, header, block) {
 73     // パラメータの調整
 74     if(header === undefined) {
 75       header = [];
 76     }
 77 
 78     // リクエストの準備
 79     var xhr = new ActiveXObject("Microsoft.XMLHTTP");
 80     xhr.open("GET", "http://" + this.host + ":" + this.port + path, false);
 81 
 82     for(var headerName in header) {
 83       xhr.setRequestHeader(headerName, header[headerName]);
 84     }
 85 
 86     // リクエストの送信
 87     xhr.send();
 88 
 89     // レスポンスの処理
 90     if(xhr.readyState == 4) {
 91       if(xhr.status == 200) {
 92         if(block) {
 93           block(xhr.responseText);
 94         } else {
 95           return xhr.responseText;
 96         }
 97       } else {
 98         throw new Error(xhr.statusText);
 99       }
100     }
101   }
102 };
