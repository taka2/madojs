1 /** 
  2  * 新しいADOのコネクションを作成する
  3  * @class ADOを使ったAccessデータベースへの接続を行うクラス
  4 <pre class = "code">
  5 使用例：
  6 AdoAccessConnection.open("c:\\path\\to\\test.mdb", "hoge", "fuga", function(con) {
  7   var result = con.executeQuery("SELECT * FROM TEST");
  8 
  9   result.each(function(row) {
 10     print(row["COL1"]); // 'X'
 11   });
 12 });
 13 
 14 </pre>
 15  * @param {String} mdbFilePath mdbファイルのパス
 16  * @param {String} userName ユーザ名
 17  * @param {String} password パスワード
 18  */
 19 var AdoAccessConnection = function(mdbFilePath, userName, password) {
 20   var connectString = "Provider=Microsoft.Jet.OLEDB.4.0;Data Source=" + mdbFilePath;
 21   this.con = new AdoConnection(connectString, userName, password).getRawConnection();
 22 };
 23 
 24 /** 
 25  * 新しいADOのコネクションを作成し、ブロックを実行します。
 26  * ブロックが指定されていない場合は、作成したADOのコネクションを返します。
 27  * @param {String} mdbFilePath mdbファイルのパス
 28  * @param {String} userName ユーザ名
 29  * @param {String} password パスワード
 30  * @param {Function} block ブロック
 31  * @return {Object} ブロックが指定されていない場合は、作成したADOのコネクション
 32  */
 33 AdoAccessConnection.open = function(mdbFilePath, userName, password, block) {
 34   if (!isFunction(block)) {
 35     return new AdoAccessConnection(mdbFilePath, userName, password);
 36   }
 37 
 38   try {
 39     var con = new AdoAccessConnection(mdbFilePath, userName, password);
 40     block(con);
 41   } finally {
 42     if(con) {
 43       con.close();
 44     }
 45   }
 46 };
 47 
 48 // Prototypes of AdoAccessConnection
 49 AdoAccessConnection.prototype = Object.create(AdoConnection.prototype);
 50 AdoAccessConnection.prototype.constructor = AdoAccessConnection;
