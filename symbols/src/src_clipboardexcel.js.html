1 /** 
  2  * 新しいクリップボードを作成する
  3  * @class Excelを利用した、クリップボードへの格納、および、取得を行うクラス。
  4  */
  5 var ClipboardExcel = function() {
  6   this.excel = Excel.create();
  7   this.sheet = this.excel.getSheetByIndex(0);
  8   this.sheet.setFormat(1, 1, "@");
  9 };
 10 
 11 /** 
 12  * 新しいクリップボードを作成し、ブロックを実行します。
 13  * ブロックが指定されていない場合は、作成したクリップボードを返します。
 14  * @param {Function} block ブロック
 15  * @return {Object} ブロックが指定されていない場合は、作成したクリップボード
 16  */
 17 ClipboardExcel.open = function(block) {
 18   if (!isFunction(block)) {
 19     return new ClipboardExcel();
 20   }
 21 
 22   try {
 23     var clip = new ClipboardExcel();
 24     block(clip);
 25   } finally {
 26     if(clip) {
 27       clip.close();
 28     }
 29   }
 30 };
 31 
 32 // Prototypes of ClipboardExcel
 33 ClipboardExcel.prototype = {
 34   /**
 35    * クリップボードへ文字列を格納します。
 36    * @param {String} text 格納する文字列
 37    */
 38   set: function(text) {
 39     this.sheet.setValue(1, 1, text);
 40     this.sheet.copy(1, 1);
 41   },
 42   /**
 43    * クリップボードの文字列を取得します。
 44    * @return {String} 取得した文字列
 45    */
 46   get: function() {
 47     return this.sheet.getValue(1, 1);
 48   },
 49   /**
 50    * クリップボードをクローズします。
 51    */
 52   close: function() {
 53     this.excel.quitDiscardChanges();
 54   }
 55 };
