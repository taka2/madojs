1 // Constructor of File
  2 var File = function(path, mode) {
  3   this.path = path;
  4   this.mode = mode;
  5 
  6   if(this.mode === 'w') {
  7     this.ts = Const.FSO.CreateTextFile(this.path);
  8   } else if(this.mode === 'a') {
  9     if(File.exist(this.path)) {
 10       this.ts = Const.FSO.OpenTextFile(this.path, Const.IOMODE_FORAPPENDING);
 11     } else {
 12       this.ts = Const.FSO.CreateTextFile(this.path);
 13     }
 14   } else {
 15     this.ts = Const.FSO.OpenTextFile(this.path, Const.IOMODE_FORREADING);
 16   }
 17 };

 19 // Static methods of File
 20 File.open = function(path, mode, block) {
 21   var file = new File(path, mode);
 22   if(block) {
 23     try {
 24       block(file);
 25     } finally {
 26       file.close();
 27     }
 28   } else {
 29     return file;
 30   }
 31 };

 33 File.rename = function(from, to) {
 34   if(Const.FSO.FolderExists(from)) {
 35     Const.FSO.MoveFolder(from, to);
 36   } else if(Const.FSO.FileExists(from)) {
 37     Const.FSO.MoveFile(from, to);
 38   } else {
 39     throw new Error(-1, "File Not Found: " + from);
 40   }
 41 };
 42 
 43 File.copy = function(from, to) {
 44   if(Const.FSO.FolderExists(from)) {
 45     Const.FSO.CopyFolder(from, to);
 46   } else if(Const.FSO.FileExists(from)) {
 47     Const.FSO.CopyFile(from, to);
 48   } else {
 49     throw new Error(-1, "File Not Found: " + from);
 50   }
 51 };

 53 File.exist = function(path) {
 54   if(Const.FSO.FolderExists(path)) {
 55     return true;
 56   } else if(Const.FSO.FileExists(path)) {
 57     return true;
 58   } else {
 59     return false;
 60   }
 61 };

 64 File.unlink = function(path, force) {
 65   if(!force) {
 66     force = false;
 67   }
 68   if(Const.FSO.FolderExists(path)) {
 69     Const.FSO.DeleteFolder(path, force);
 70   } else if(Const.FSO.FileExists(path)) {
 71     Const.FSO.DeleteFile(path, force);
 72   } else {
 73     throw new Error(-1, "File Not Found: " + path);
 74   }
 75 };

 77 File.size = function(path) {
 78   if(Const.FSO.FolderExists(path)) {
 79     return Const.FSO.GetFolder(path).Size;
 80   } else if(Const.FSO.FileExists(path)) {
 81     return Const.FSO.GetFile(path).Size;
 82   } else {
 83     throw new Error(-1, "File Not Found: " + path);
 84   }
 85 };

 87 File.realpath = function(pathname, basedir) {
 88   if(!basedir) {
 89     basedir = Dir.getwd();
 90   }
 91 
 92   Dir.cd(basedir, function(path) {
 93     realpath = Const.FSO.GetAbsolutePathName(pathname);
 94   });
 95 
 96   return realpath;
 97 };

 99 File.extname = function(filename) {
100   var extensionName = Const.FSO.GetExtensionName(filename);
101   if(extensionName === "") {
102     return extensionName;
103   } else {
104     return "." + extensionName;
105   }
106 };
107 
108 File.basename = function(filename) {
109   var file = Const.FSO.GetFile(filename);
110   return file.Name;
111 };

113 // Prototypes of File
114 File.prototype = {
115   each: function(block) {
116     while(!this.ts.AtEndOfStream) {
117       block(this.ts.ReadLine());
118     }
119   },
120   puts: function(line) {
121     this.ts.WriteLine(line);

124     this.ts.Close();
