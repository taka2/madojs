1 /** 
  2  * 新しいキー送信クラスを作成する。
  3  * @class 連続で自動的にキー送信を行う
  4 <pre class = "code">
  5 使用例：
  6 // ノートパッドを起動
  7 Process.exec("notepad", [], Const.WINDOW_STYLE_NORMAL, false);
  8 
  9 // 起動を待つ
 10 sleep(100);
 11 
 12 // "日本語"という文字列を書きこんで、abc.txtに保存。
 13 var ks = new KeySender("無題")
 14     .sendJapaneseKeys("日本語")
 15     .sendKeyWithControl("s")
 16     .sendKeys("abc.txt")
 17     .sendEnter();
 18 </pre>
 19  */
 20 var KeySender = function(targetWindow) {
 21   Const.WSHELL.AppActivate(targetWindow);
 22   sleep(KeySender.DURATION);
 23 };
 24 
 25 // Constants of KeySender
 26 // private
 27 // AppActivateしてからKeySendするまでの待ち時間
 28 KeySender.DURATION = 10;

 36    * @return this
 37    */

 39     sendKeys(text, num);
 40     return this;
 41   },

 46    * @return this
 47    */
 48   sendJapaneseKeys: function(text, num) {
 49     Clipboard.open(function(clip) {
 50       for(var i=0; i<num; i++) {
 51         clip.set(text);
 52         sendKeys("^v");
 53       }
 54     });
 55     return this;
 56   },
