1 /**
  2  * インスタンス化しません。
  3  * @class JSONの解析(parse)と、文字列化(stringify)をサポートするクラス
  4 <pre class = "code">
  5 使用例：
  6 JSON.parse("[1,2,3]");
  7 JSON.stringify([1,2,3]);
  8 </pre>
  9  */
 10 var JSON = {};
 11 
 12 /**
 13  * 指定した文字列をオブジェクトを解析してオブジェクトとして返します。
 14  * @param {String} str JSON文字列
 15  * @return {Object} 解析結果のオブジェクト
 16  */
 17 JSON.parse = function(str) {
 18   return eval('(' + str + ')');
 19 };
 20 
 21 /**
 22  * 指定したオブジェクトを文字列として返します。
 23  * @param {String} obj オブジェクト
 24  * @return {Object} 文字列化したオブジェクト
 25  */
 26 JSON.stringify = function(obj) {
 27   return obj.toString();
 28 };
