1 /** 
  2  * 文字列先頭と末尾のスペースをトリムした文字列を生成して返します。
  3  * @return {String} トリムした文字列
  4  * @example 使用例：
  5 print("@" + "   	abc			".trim() + "@"); // "@abc@"
  6  */
  7 String.prototype.trim = function() {
  8   var text = this.valueOf();
  9   return (text || "").replace( /^\s+|\s+$/g, "" );
 10 };
 11 
 12 /**
 13  * 文字列がstrで始まるかどうか
 14  * @param {String} str テストする文字列
 15  * @return {Boolean} 文字列がstrで始まる場合true、そうでない場合はfalseを返す。
 16  * @example 使用例：
 17 "abc".startsWith("a"); // true
 18  */
 19 String.prototype.startsWith = function(str) {
 20   var text = this.valueOf();
 21   return(text.match("^" + str) === null ? false : true);
 22 };
 23 
 24 /**
 25  * 文字列がstrで終わるかどうか
 26  * @param {String} str テストする文字列
 27  * @return {Boolean} 文字列がstrで終わる場合true、そうでない場合はfalseを返す。
 28  * @example 使用例：
 29 "abc".endsWith("c"); // true
 30  */
 31 String.prototype.endsWith = function(str) {
 32   var text = this.valueOf();
 33   return(text.match(str + "$") === null ? false : true);
 34 };
 35 
 36 /**
 37  * chが全角かどうか判定する。<br/>
 38  * 二文字以上の文字列でも、先頭の文字のみ判定します。
 39  * @return {Boolean} chが全角の場合true、そうでない場合falseを返す。
 40  */
 41 String.prototype.isZenkaku = function() {
 42   // 先頭一文字のみチェックする
 43   var text = this.valueOf();
 44   var c = text.substring(0, 1);
 45 
 46   // escapeした結果%u始まり、かつ、半角カナでない場合全角と判定する
 47   return (escape(c).substring(0, 2) === "%u") &&
 48           !c.isHankakuKana();
 49 }
 50 
 51 /**
 52  * chが半角カナかどうか判定する。<br/>
 53  * 二文字以上の文字列でも、先頭の文字のみ判定します。
 54  * @return {Boolean} chが半角カナの場合true、そうでない場合falseを返す。
 55  */
 56 String.prototype.isHankakuKana = function() {
 57   // 先頭一文字のみチェックする
 58   var text = this.valueOf();
 59   var c = text.substring(0, 1);
 60   var code = c.charCodeAt(0);
 61 
 62   return(code >= 0xFF71 && code <= 0xFF90);
 63 }

 65 /**
 66  * 全角文字を2バイト、半角文字を1バイトとして、指定位置の文字列を返します。
 67  * @param {Number} start 開始位置
 68  * @param {Number} end 終了位置
 69  * @return {String} 指定位置の文字列
 70  */
 71 String.prototype.substringb = function(start, end) {
 72   var text = this.valueOf();
 73   var currentIndex = 0;
 74   var result = "";
 75 
 76   var textLength = text.length;
 77   for(var i=0; i<textLength; i++) {
 78     var ch = text.charAt(i);
 79     var chSize = 1;
 80     if(ch.isZenkaku()) {
 81       chSize = 2;
 82     }
 83     if(currentIndex >= start && (currentIndex + chSize) <= end) {
 84       result += ch;
 85     }
 86 
 87     currentIndex += chSize;
 88   }
 89 
 90   return result;
 91 }

 93 /**
 94  * 指定文字を、指定文字数に達するまで左側にパディングします。
 95  * @param {String} ch 文字
 96  * @param {Number} length 文字数
 97  * @return {String} 指定位置の文字列
 98  * @example 使用例：
 99 "12".lpad("0", 5); // "00012"
100  */
101 String.prototype.lpad = function(ch, length) {
102   var text = this.valueOf();
103 
104   // 文字列が指定文字数に既に達している場合はそのまま返す。
105   if(text.length >= length) {
106     return text;
107   }
108 
109   var result = text;
110   for(var i=0; i<length - text.length; i++) {
111     result = ch + result;
112   }
113   return result;
114 }
