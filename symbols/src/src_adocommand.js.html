1 /** 
  2  * 新しいADOのコマンドを作成する。<br/>
  3    通常、AdoCommandを直接インスタンス化せず、AdoConnectionのサブクラスから、createCommandを呼びます。
  4  * @class ADOコマンドを表すクラス。クエリにパラメータが必要な場合は、このクラスを使います。
  5 <pre class = "code">
  6 使用例：
  7 AdoAccessConnection.open("c:\\path\\to\\test.mdb", "", "", function(con) {
  8   var command = con.createCommand("UPDATE TEST SET COL2 = ? WHERE COL1 = ?");
  9   command.setStringParameter("COL2", "えええ");
 10   command.setStringParameter("COL1", "Z");
 11   command.executeUpdate();
 12 });
 13 
 14 AdoAccessConnection.open("c:\\path\\to\\test.mdb", "", "", function(con) {
 15   var command = con.createCommand("SELECT COL2 FROM TEST WHERE COL1 = ?");
 16   command.setStringParameter("COL1", "Z");
 17   var rs = command.executeQuery();
 18 
 19   print(rs[0]["COL2"]);
 20 });
 21 
 22 </pre>
 23  * @param {Object} con Adoコネクション
 24  */
 25 var AdoCommand = function(con, sql) {
 26   this.con = con;
 27   this.sql = sql;
 28   this.params = [];
 29 
 30   // コマンドの生成
 31   this.command = new ActiveXObject("ADODB.Command");
 32   this.command.CommandText = this.sql;
 33   this.command.CommandType = AdoCommand.COMMAND_TYPE_CMD_TEXT;
 34   this.command.ActiveConnection = this.con.getRawConnection();
 35 };
 36

 38  * Command Typeの定数：テキスト(1)

 97 // Prototypes of AdoCommand
 98 AdoCommand.prototype = {
 99   /**
100    * 指定されたsqlを実行し、ハッシュ{fieldName: value}の配列として値を返す。
101    * @param {String} sql 実行するSQL
102    * @param {Array} params パラメータ
103    * @return {Array} ハッシュ{fieldName: value}の配列
104    */
105   executeQuery: function(sql, params) {
106     try {
107       // パラメータの追加
108       this.prepareParameter();
109 
110       // SQLの実行
111       var rs = this.command.Execute();
112       return this.con.convertToArrayRs(rs);
113     } catch(e) {
114       throw e;
115     } finally {
116       if(rs) {
117         rs.Close();
118       }
119     }
120   },
121   /**
122    * 指定されたsqlを実行する。
123    * @param {String} sql 実行するSQL
124    * @param {Array} params パラメータ
125    */
126   executeUpdate: function(sql, params) {
127     // パラメータの追加
128     this.prepareParameter();
129 
130     // SQLの実行
131     this.command.Execute();
132   },
133   // private
134   // パラメータの準備をする。
135   prepareParameter: function() {
136     // パラメータの追加
137     var command = this.command;
138     this.params.each(function(param) {
139       var objParam = command.CreateParameter(
140         param.Name
141         , param.Type
142         , param.Direction
143         , param.Size
144         , param.Value
145       );
146       command.Parameters.Append(objParam);
147     });
148   },
149   /**
150    * 論理型のパラメータを作成する。
151    * @param {String} paramName パラメータ名
152    * @param {Boolean} paramValue パラメータ値
153    */
154   setBooleanParameter: function(paramName, paramValue) {
155     this.params.push({
156       "Name": paramName
157       , "Type": AdoCommand.DATA_TYPE_BOOLEAN
158       , "Direction": AdoCommand.PARAMETER_DIRECTION_INPUT
159       , "Size": -1
160       , "Value": paramValue
161     });
162   },
163   /**
