1 /** 
  2  * pathの存在チェックを行った上で、ディレクトリオブジェクトを作成します。
  3  * @class ディレクトリの操作を行うためのクラスです。
  4  * @param  {String} path ディレクトリのパスを文字列で指定します。
  5  * @throws ディレクトリが存在しない場合にスローされます。
  6  */
  7 var Dir = function(path) {
  8   if(!Dir.exist(path)) {
  9     throw new Error(path + " not found or is not directory.");
 10   }
 11   this.path = path;
 12 };
 13 
 14 /**
 15  * カレントディレクトリのフルパスを文字列で返します。
 16  * @return {String} カレントディレクトリのフルパス
 17  */
 18 Dir.getwd = function() {
 19   return Const.WSHELL.CurrentDirectory;
 20 };

 22 /**
 23  * <a href = "#.getwd">Dir#getwd</a>のエイリアスです。
 24  * @function
 25  */
 26 Dir.pwd = Dir.getwd;

 28 /**
 29  * カレントディレクトリをpathに変更します。
 30  * ブロックが指定された場合、カレントディレクトリの変更はブロックの実行中に限られます。
 31  * @param {String} path ディレクトリのパスを文字列で指定します。
 32  * @param {Function} block ブロック
 33  */
 34 Dir.chdir = function(path, block) {
 35   if(!path) {
 36     path = Const.INITIAL_CURRENT_DIRECTORY;
 37   } else if(!File.exist(path)) {
 38     throw new Error(-1, "Path Not Found: " + path);
 39   }
 40 
 41   if(block) {
 42     // 現在のワーキングディレクトリを取得
 43     var cwd = Dir.getwd();
 44 
 45     // 指定のパスにディレクトリ移動
 46     Const.WSHELL.CurrentDirectory = path;
 47 
 48     // ブロックを実行
 49     block(path);
 50 
 51     // ワーキングディレクトリを元に戻す
 52     Const.WSHELL.CurrentDirectory = cwd;
 53   } else {
 54     Const.WSHELL.CurrentDirectory = path;
 55   };
 56 };

 58 /**
 59  * pathで指定された新しいディレクトリを作ります。
 60  * @param {String} path ディレクトリのパスを文字列で指定します。
 61  * @return {Object} 作成したディレクトリのディレクトリオブジェクト
 62  */
 63 Dir.mkdir = function(path) {
 64   Const.FSO.CreateFolder(path);
 65   return new Dir(path);
 66 };

 68 /**
 69  * pathで指定されたディレクトリの親ディレクトリのパスを文字列で取得します。
 70  * @param {String} path ディレクトリのパスを文字列で指定します。
 71  * @return {String} 親ディレクトリのパス
 72  */
 73 Dir.getParentDir = function(path) {
 74   return Const.FSO.GetParentFolderName(path);
 75 };

 77 Dir.open = function(path, block) {
 78   if(block) {
 79     var dir = new Dir(path);
 80     block(dir);
 81   } else {
 82     return new Dir(path);
 83   }
 84 };

 86 /**
 87  * file_nameで与えられたディレクトリが存在するかどうかをチェックします。
 88  * @param {String} file_name ディレクトリのパスを文字列で指定します。
 89  * @return {Boolean} 存在する場合はtrue、存在しない場合はfalseを返します。
 90  */
 91 Dir.exist = function(file_name) {
 92   return Const.FSO.FolderExists(file_name);
 93 };

 95 /**
 96  * ディレクトリpathに含まれるファイルエントリ名の配列を返します。
 97  * @param {String} path ディレクトリのパスを文字列で指定します。
 98  * @return {Boolean} ファイルエントリ名の配列
 99  */
100 Dir.entries = function(path) {
101   var dir = new Dir(path);
102 
103   var result = [];
104 
105   dir.each(function(item) {
106     result.push(item);
107   });
108 
109   return result;
110 };

112 /**
113  * ディレクトリpathに含まれる(サブディレクトリも含む)ファイルエントリ名の配列を返します。
114  * patternを指定した場合、patternの正規表現にマッチするファイルのみ抽出します。
115  * @param {String} path ディレクトリのパスを文字列で指定します。
116  * @param {String} pattern 抽出するファイル名の正規表現パターン
117  * @param {Function} block ブロック
118  * @return {Boolean} ファイルエントリ名の配列
119  */
120 Dir.find = function(path, pattern, block) {
121   var dir = new Dir(path);
122 
123   var result = [];
124 
125   var processFile = function(item) {
126     if(pattern === undefined || item.match(pattern)) {
127       if(block) {
128         block(item);
129       } else {
130         result.push(item);
131       }
132     }
133   };
134 
135   dir.each(function(item) {
136     if(Dir.exist(item)) {
137       var subresult = Dir.find(item, pattern, block);
138       for(var i=0; i<subresult.length; i++) {
139         processFile(subresult[i]);
140       }
141     } else {
142       processFile(item);
143     }
144   });
145 
146   return result;
147 };

149 // Prototypes of Dir
150 Dir.prototype = {
151   /**
152    * このオブジェクトが持つpathで指定されたディレクトリの親ディレクトリのパスを文字列で取得します。
153    * @return {String} 親ディレクトリのパス
154    */
155   getParentDir: function() {
156     return Dir.getParentDir(this.path);
157   },
158   /**
159    * このオブジェクトが持つpathに含まれるファイルエントリ名の配列を返します。
160    * @return {Boolean} ファイルエントリ名の配列
161    */
162   each: function(block) {
163     var dir = Const.FSO.GetFolder(this.path);
164 
165     var enumDir = new Enumerator(dir.SubFolders);
166     for (;!enumDir.atEnd(); enumDir.moveNext()) {
167        block(enumDir.item().Path);
168     }
169 
170     var enumFile = new Enumerator(dir.Files);
171     for (;!enumFile.atEnd(); enumFile.moveNext()) {
172        block(enumFile.item().Path);
173     }
174   }
175 };
