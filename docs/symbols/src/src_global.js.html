  2 /**
  3  * 指定したmsec(ミリ秒)スリープします。
  4  * @param {Number} msec スリープするミリ秒
  5  */
  6 var sleep = function(msec) {
  7   WScript.Sleep(msec);
  8 };
  9 
 10 /**
 11  * 指定したfnがtrueを返すまで、msec(ミリ秒)間隔でfnを実行します。
 12  * @param {Number} msec スリープするミリ秒
 13  * @param {Function} fn 実行する関数
 14  * @param {Array} args 関数に渡す引数
 15  */
 16 var sleepif = function(msec, fn, args) {
 17   while(true) {
 18     if(fn(args)) {
 19       return true;
 20     }
 21     sleep(msec);
 22   }
 23 };

 85 /**
 86  * OSをシャットダウンします。
 87  * @param {Number} timeout タイムアウトを秒で指定
 88  */
 89 var osShutdown = function (timeout) {
 90   if (timeout === null) timeout = 0;
 91   Process.exec("shutdown", ["-s", "-t", timeout], 0, false);
 92 };
 93 
 94 /**
 95  * OSを再起動します。
 96  * @param {Number} timeout タイムアウトを秒で指定
 97  * @param {String} option "-f"を指定すると、実行中のプロセスを警告なしで閉じます
 98  */
 99 var osReboot = function (timeout, option) {
100   var params = ["-r", "-t"];
101   if (timeout === null) timeout = 0;
102   params.push(timeout);
103 
104   if (option !== null) {
105     params.push(option);
106   }
107 
108   Process.exec("shutdown", params, 0, false);
109 };

111 /**
112  * ショートカットを作成します。
113  * @param {String} pathTo 作成するショートカットの場所
114  * @param {String} pathFrom ショートカットの対象となるファイル、または、URL
115  * @example 使用例：
116 // SendToにノートパッドのショートカットを作成
117 createShortcut(SpecialFolders.getSendTo() + "\\notepad.lnk", "notepad.exe");
118 
119 // デスクトップにYahooのURLショートカットを作成
120 createShortcut(SpecialFolders.getDesktop() + "\\yahoo.url", "http://www.yahoo.co.jp/");
121  */
122 var createShortcut = function(pathTo, pathFrom) {
123   var shortcut = Const.WSHELL.CreateShortcut(pathTo);
124   shortcut.TargetPath = pathFrom;
125   shortcut.Save();
126 };
