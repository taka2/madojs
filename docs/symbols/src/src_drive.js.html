1 /** 
  2  * ドライブの存在チェックを行った上で、ドライブオブジェクトを作成します。
  3  * @class ドライブの操作を行うためのクラスです。
  4 <pre class = "code">
  5 使用例：
  6 // Cドライブの空き容量を表示
  7 var d = new Drive("c:");
  8 print(d.getFreeSpace());
  9 </pre>
 10  * @param  {String} path ディレクトリのパスを文字列で指定します。
 11  * @throws ディレクトリが存在しない場合にスローされます。
 12  */
 13 var Drive = function(path) {
 14   if(!Drive.exist(path)) {
 15     throw new Error("Drive " + path + " not found.");
 16   }
 17   this.path = path;
 18   this.driveObj = Const.FSO.GetDrive(path);
 19 };

 51 /**
 52  * ドライブの一覧を取得します。
 53  * @return {Array} Driveオブジェクトの配列
 54  */
 55 Drive.getDrives = function() {
 56   var result = [];
 57   var e = new Enumerator(Const.FSO.Drives);
 58   for (; !e.atEnd(); e.moveNext())
 59   {
 60     result.push(new Drive(e.item().Path));
 61   }
 62 
 63   return result;
 64 };
 65 
 66 /**
 67  * pathで与えられたドライブが存在するかどうかをチェックします。
 68  * @param {String} path ドライブのパスを文字列で指定します。
 69  * @return {Boolean} 存在する場合はtrue、存在しない場合はfalseを返します。
 70  */
 71 Drive.exist = function(path) {
 72   return Const.FSO.DriveExists(path);
 73 };
